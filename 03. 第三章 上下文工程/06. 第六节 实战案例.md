## 案例一：企业知识库 RAG 系统

### 需求背景

某科技公司有大量内部文档（产品手册、技术文档、会议纪要、规章制度），员工经常需要在这些文档中查找信息。

现状问题：
- 文档太多，人工查找效率低
- 新员工不了解公司文档结构，不知道去哪找
- 有些文档已经过时，但员工不知道，用了错误的信息

目标：构建一个基于 RAG 的企业知识库问答系统。

### 系统架构

```
用户问题
    ↓
┌─────────────────────────────┐
│  预处理层                   │
│  - 查询改写                 │
│  - 意图识别                 │
│  - 查询扩展                 │
└─────────────────────────────┘
    ↓
┌─────────────────────────────┐
│  检索层                     │
│  - 混合检索（向量+关键词）   │
│  - 重排序                   │
│  - 过滤（时效、权限、部门）  │
└─────────────────────────────┘
    ↓
┌─────────────────────────────┐
│  生成层                     │
│  - 上下文组装               │
│  - Prompt 构建              │
│  - 生成回答                 │
└─────────────────────────────┘
    ↓
用户回答（含引用来源）
```

### 1. 文档处理与索引

首先，需要把企业文档处理成适合检索的格式。

```python
from langchain.document_loaders import DirectoryLoader
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain.embeddings import OpenAIEmbeddings
from langchain.vectorstores import Chroma

# 1. 加载文档
loader = DirectoryLoader(
    "./docs",  # 文档目录
    glob="**/*.md",  # 只处理 Markdown 文件
    show_progress=True
)
documents = loader.load()

# 2. 文档预处理
processed_docs = []
for doc in documents:
    # 添加元数据
    doc.metadata.update({
        "source": "企业知识库",
        "department": extract_department(doc.metadata.get("source", "")),
        "created_at": parse_date(doc.metadata.get("created", "")),
        "last_updated": parse_date(doc.metadata.get("updated", "")),
        "doc_type": classify_doc_type(doc.metadata.get("source", ""))
    })

    # 过滤过时文档（超过1年未更新的）
    if is_stale(doc.metadata["last_updated"], threshold_days=365):
        continue

    processed_docs.append(doc)

# 3. 分块
text_splitter = RecursiveCharacterTextSplitter(
    chunk_size=1000,  # 每个块约 1000 tokens
    chunk_overlap=200,  # 相邻块重叠 200 tokens，保持上下文连续
    separators=["\n\n", "\n", ". ", "，"]  # 优先按段落分割
)

chunks = text_splitter.split_documents(processed_docs)

# 4. 向量化并存储
embeddings = OpenAIEmbeddings(model="text-embedding-3-small")

vectorstore = Chroma.from_documents(
    documents=chunks,
    embedding=embeddings,
    persist_directory="./vector_db"  # 持久化存储
)

print(f"索引完成：{len(chunks)} 个文档块")
```

### 2. 查询预处理

用户的问题往往不够精确，需要预处理提升检索质量。

```python
def preprocess_query(user_query, user_context=None):
    """
    查询预处理
    """
    # 1. 查询改写：让查询更精确
    rewritten = rewrite_query(user_query)

    # 2. 查询扩展：补充相关概念
    expanded = expand_query(rewritten)

    # 3. 意图识别：判断用户想要什么
    intent = classify_intent(user_query)

    # 4. 根据意图调整查询
    if intent == "how_to":
        # 问"怎么做"，扩展为包含"步骤""教程"等
        expanded = f"{expanded} 步骤 教程 方法"
    elif intent == "why":
        # 问"为什么"，扩展为包含"原因""原理"
        expanded = f"{expanded} 原因 为什么 原理"
    elif intent == "what":
        # 问"是什么"，扩展为包含"定义""概念"
        expanded = f"{expanded} 定义 概念 是什么"

    return {
        "original": user_query,
        "rewritten": rewritten,
        "expanded": expanded,
        "intent": intent
    }


def rewrite_query(query):
    """
    查询改写：大模型帮我们把模糊查询改写成精确查询
    """
    prompt = f"""
    请将以下用户查询改写得更精确、更适合检索。

    原查询：{query}

    要求：
    1. 去除口语化表达
    2. 补充隐含的上下文
    3. 使用更精确的术语
    4. 如果原查询已经很精确，保持不变

    改写后的查询：
    """
    response = llm.generate(prompt)
    return response.strip()
```

### 3. 混合检索与重排序

```python
def hybrid_search(query, filters=None, top_k=10):
    """
    混合检索：向量搜索 + BM25 关键词搜索
    """
    # 向量检索
    vector_results = vectorstore.similarity_search_with_score(
        query,
        k=top_k * 2  # 多召回一些，后面会过滤
    )

    # BM25 关键词检索
    bm25_results = bm25_index.search(query, k=top_k * 2)

    # 结果融合（RRF）
    fused = reciprocal_rank_fusion(
        [vector_results, bm25_results],
        k=60  # RRF 常数
    )

    # 应用过滤器
    if filters:
        fused = apply_filters(fused, filters)

    # 取 top_k
    return fused[:top_k]


def apply_filters(results, filters):
    """
    应用过滤条件
    """
    filtered = []
    for doc, score in results:
        # 时效过滤
        if "min_freshness" in filters:
            if doc.metadata.get("last_updated") < filters["min_freshness"]:
                continue

        # 部门过滤
        if "department" in filters:
            if doc.metadata.get("department") != filters["department"]:
                continue

        # 文档类型过滤
        if "doc_type" in filters:
            if doc.metadata.get("doc_type") not in filters["doc_type"]:
                continue

        filtered.append((doc, score))

    return filtered


def rerank(query, results, top_k=5):
    """
    重排序：用更大的模型对检索结果排序
    """
    from sentence_transformers import CrossEncoder

    cross_encoder = CrossEncoder("cross-encoder/ms-marco-MiniLM-L-12-v2")

    # 准备查询-文档对
    pairs = [(query, doc.page_content) for doc, _ in results]

    # 获取相关性分数
    scores = cross_encoder.predict(pairs)

    # 按分数排序
    reranked = sorted(zip(results, scores), key=lambda x: x[1], reverse=True)

    return [doc for doc, _ in reranked[:top_k]]
```

### 4. 上下文组装与生成

```python
SYSTEM_PROMPT = """你是一个企业知识库问答助手。

你的职责：
1. 基于提供的文档片段回答用户问题
2. 如果文档中没有相关信息，直接说"我没有找到相关信息"
3. 回答时必须引用你参考的文档
4. 如果多个文档提供不同信息，指出这一点

回答要求：
- 简洁、准确
- 用中文回答
- 如果涉及步骤，要有清晰的序号
- 如果有多个可能的答案，列出它们并说明适用情况

你只能看到【参考文档】部分的内容，不要编造信息。"""


USER_PROMPT_TEMPLATE = """【用户问题】
{user_question}

【参考文档】
{context}

【回答要求】
请基于参考文档回答用户问题。
如果需要，可在回答末尾添加"参考来源"部分。"""


def assemble_context(docs, max_tokens=4000):
    """
    将检索到的文档组装成上下文
    """
    context_parts = []
    current_tokens = 0

    for i, doc in enumerate(docs, 1):
        doc_tokens = count_tokens(doc.page_content)

        if current_tokens + doc_tokens > max_tokens:
            # 剩余空间不够放完整文档，尝试放摘要
            remaining = max_tokens - current_tokens
            if remaining > 150:
                summary = summarize(doc.page_content, max_tokens=remaining)
                context_parts.append(f"[文档{i}摘要] {summary}")
                current_tokens += count_tokens(summary)
            break

        # 添加完整文档，标注来源
        source = doc.metadata.get("source", "未知来源")
        section = doc.metadata.get("section_title", "")
        context_parts.append(
            f"【文档{i} | 来源：{source} | 章节：{section}】\n{doc.page_content}"
        )
        current_tokens += doc_tokens

    return "\n\n".join(context_parts)


def answer_question(user_query, user_filters=None):
    """
    完整的问答流程
    """
    # 1. 查询预处理
    processed = preprocess_query(user_query)

    # 2. 检索（带用户可能的部门过滤）
    filters = {}
    if user_filters:
        filters.update(user_filters)
    docs = hybrid_search(processed["expanded"], filters=filters)

    if not docs:
        return "我没有找到与你的问题相关的文档。"

    # 3. 重排序
    docs = rerank(processed["expanded"], docs)

    # 4. 组装上下文
    context = assemble_context(docs)

    # 5. 构建 Prompt
    user_prompt = USER_PROMPT_TEMPLATE.format(
        user_question=user_query,
        context=context
    )

    # 6. 生成回答
    response = llm.generate(
        system=SYSTEM_PROMPT,
        user=user_prompt
    )

    return response
```

```python
# 测试用例
test_cases = [
    {
        "query": "怎么申请年假？",
        "expected_topics": ["年假", "申请流程", "请假制度"]
    },
    {
        "query": "我们的代码规范是什么？",
        "expected_topics": ["代码规范", "编码规范", "最佳实践"]
    },
    {
        "query": "这台服务器是怎么配置的？",
        "expected_topics": ["服务器配置", "运维"]
    }
]

# 评估函数
def evaluate_rag(query, expected_topics):
    docs = hybrid_search(query, top_k=5)
    retrieved_topics = set()
    for doc in docs:
        retrieved_topics.update(extract_topics(doc.page_content))

    # 计算召回率
    recall = len(expected_topics & retrieved_topics) / len(expected_topics)

    return {
        "query": query,
        "expected": expected_topics,
        "retrieved": list(retrieved_topics),
        "recall": recall
    }

# 运行评估
for test in test_cases:
    result = evaluate_rag(test["query"], test["expected_topics"])
    print(f"查询：{result['query']}")
    print(f"召回率：{result['recall']:.2%}")
```

## 案例二：Agent 用户画像与记忆系统

### 需求背景

一个 AI 个人助理，需要记住用户的信息和偏好，提供个性化的服务。

需要记忆的信息：
- 用户基本信息（姓名、职业、使用偏好）
- 项目背景（用户当前在做什么项目）
- 交互习惯（用户喜欢什么格式的回答）
- 长期知识（用户学到的概念、记住的事情）

### 记忆系统设计

```python
from datetime import datetime, timedelta
from typing import Dict, List, Optional

class Memory:
    """Agent 的记忆系统"""

    def __init__(self):
        # 各类记忆存储
        self.short_term = []      # 短期记忆：当前任务的中间状态
        self.conversation = []     # 对话记忆：当前对话的历史
        self.user_profile = {}     # 用户画像：基本信息
        self.project_context = {}  # 项目上下文：当前项目背景
        self.knowledge = []       # 知识记忆：用户学到的概念
        self.preferences = {}      # 偏好：用户的交互偏好

        # 记忆存储后端（可用 Redis、SQLite 等）
        self.storage = MemoryStorage()

    def save(self, memory_type: str, content: str, metadata: dict = None):
        """保存记忆"""
        memory_entry = {
            "id": generate_id(),
            "type": memory_type,
            "content": content,
            "metadata": metadata or {},
            "created_at": datetime.now(),
            "importance": self._evaluate_importance(content, memory_type),
            "access_count": 0,
            "last_accessed": datetime.now()
        }

        self.storage.add(memory_entry)

        # 更新内存缓存
        if memory_type == "short_term":
            self.short_term.append(memory_entry)
        elif memory_type == "user_profile":
            self.user_profile.update(metadata)

    def retrieve(self, query: str, memory_types: List[str] = None,
                 top_k: int = 5) -> List[str]:
        """
        检索相关记忆
        """
        memory_types = memory_types or ["user_profile", "project_context",
                                         "knowledge", "preferences"]

        # 在存储中搜索
        results = self.storage.search(
            query=query,
            types=memory_types,
            top_k=top_k
        )

        # 更新访问记录
        for r in results:
            self.storage.update_access(r["id"])

        return [r["content"] for r in results]

    def build_context(self, query: str = None) -> str:
        """
        为当前任务构建上下文
        """
        context_parts = []

        # 1. 用户画像（始终提供）
        if self.user_profile:
            profile = "【用户信息】\n"
            for key, value in self.user_profile.items():
                profile += f"- {key}：{value}\n"
            context_parts.append(profile)

        # 2. 项目上下文（如果相关）
        if query and self._is_relevant_to_project(query):
            context_parts.append(f"【项目背景】\n{self.project_context.get('summary', '')}")

        # 3. 相关知识记忆
        if query:
            relevant_knowledge = self.retrieve(query, ["knowledge"], top_k=3)
            if relevant_knowledge:
                context_parts.append(f"【用户之前学到的】\n" + "\n".join(relevant_knowledge))

        # 4. 用户偏好
        if self.preferences:
            prefs = "【用户偏好】\n"
            for key, value in self.preferences.items():
                prefs += f"- {key}：{value}\n"
            context_parts.append(prefs)

        return "\n\n".join(context_parts) if context_parts else ""

    def _evaluate_importance(self, content: str, memory_type: str) -> float:
        """
        评估记忆的重要性（0-1）
        """
        # 显式声明的重要
        if any(kw in content for kw in ["重要", "必须记住", "关键"]):
            return 0.9

        # 用户明确提供的
        if memory_type == "user_profile":
            return 0.8

        # 项目核心信息
        if memory_type == "project_context":
            return 0.7

        # 一般知识
        if memory_type == "knowledge":
            return 0.5

        return 0.3

    def cleanup(self):
        """
        记忆清理：删除不重要的、过时的记忆
        """
        # 删除低重要性且长期未访问的记忆
        threshold = datetime.now() - timedelta(days=90)
        self.storage.delete_where(
            importance__lt=0.3,
            last_accessed__lt=threshold
        )

        # 压缩短期记忆（只保留最近的）
        if len(self.short_term) > 20:
            # 保留最重要的 20 条
            self.short_term.sort(key=lambda x: x["importance"], reverse=True)
            self.short_term = self.short_term[:20]

        # 合并过于相似的记忆
        self.storage.deduplicate(similarity_threshold=0.95)
```

### Agent 中的记忆使用

```python
class PersonalAssistant:
    """个人助理 Agent"""

    def __init__(self):
        self.memory = Memory()
        self.llm = LLM()

    def chat(self, user_message: str) -> str:
        """
        处理用户消息
        """
        # 1. 保存用户消息到对话记忆
        self.memory.save("conversation", user_message, {"role": "user"})

        # 2. 构建上下文（检索相关记忆）
        context = self.memory.build_context(user_message)

        # 3. 构建 Prompt
        prompt = f"""
{context}

【当前对话】
用户：{user_message}

请回答用户的问题或请求。
"""

        # 4. 生成回复
        response = self.llm.generate(system=SYSTEM_PROMPT, user=prompt)

        # 5. 保存 AI 回复到对话记忆
        self.memory.save("conversation", response, {"role": "assistant"})

        # 6. 检查是否需要提取重要信息到用户画像
        self._update_profile_if_needed(user_message, response)

        return response

    def _update_profile_if_needed(self, user_message: str, response: str):
        """
        检查对话中是否有重要信息需要提取保存
        """
        extract_prompt = f"""
请从以下对话中提取需要长期记忆的信息。
如果有，列出：信息类型、具体内容、重要性（高/中/低）。

对话：
用户：{user_message}
助手：{response}

需要长期记忆的信息：
"""
        result = self.llm.generate(prompt=extract_prompt)

        # 解析结果，保存到对应的记忆类型
        for item in parse_extracted_info(result):
            self.memory.save(item["type"], item["content"], {
                "importance": item["importance"],
                "source": "conversation_extraction"
            })
```

## 本节小结

两个实战案例的核心设计：

**案例一：企业知识库 RAG 系统**
- 文档处理：分块策略（元数据丰富化 + 重叠分块）
- 查询预处理：改写 + 扩展 + 意图识别
- 混合检索：向量 + BM25 + RRF 融合 + 重排序
- 上下文组装：按 Token 上限截断 + 来源标注

**案例二：Agent 记忆系统**
- 多层记忆：短期 / 对话 / 用户画像 / 项目上下文 / 知识 / 偏好
- 重要性评估：自动判断哪些值得记住
- 上下文构建：按需检索 + 智能组装
- 主动遗忘：清理低价值、过时的记忆

**核心原则**：上下文工程不是一次性设计，而是根据实际效果持续迭代优化。
