# 第六节 实战案例

## 企业级 AI 评估体系构建

这一节通过一个完整案例，展示如何从零开始为企业的 AI 应用构建评估体系。案例背景：

> **背景**：某科技公司的 AI 客服机器人，需要评估其在实际业务场景中的表现。
> **现状**：机器人上线 3 个月，用户反馈"有时答非所问"，但没有具体数据支撑。
> **目标**：建立系统化的评估体系，持续监控和优化机器人质量。

```python
            "description": "回答相关性",
            "target": 0.80,
            "measure": "回答是否与用户的问题相关"

        "completeness": {
            "description": "回答完整性",
            "target": 0.75,
            "measure": "回答是否包含用户需要的所有信息"
        },
        "clarity": {
            "description": "表达清晰度",
            "target": 0.80,
            "measure": "回答是否清晰易懂"
        },
        "safety": {
            "description": "安全性",
            "target": 0.99,  # 安全性要求极高
            "measure": "是否包含有害、错误或不当内容"
        }
    }

    # 第三层：效率指标（系统性能）
    efficiency_metrics = {
        "latency_p95": {
            "description": "P95 响应延迟（秒）",

            "measure": "95% 的请求在 3 秒内响应"

        }
    }

    def get_all_metrics(cls):
        """获取所有指标的完整定义"""
```

```python
def build_evaluation_dataset(use_cases):
    """
    构建评估数据集

    use_cases: 业务定义的用例列表
    """
    test_data = []

    for use_case in use_cases:
        # 为每个用例生成测试数据
        test_item = {
            "id": use_case["id"],
            "category": use_case["category"],
            "intent": use_case["intent"],
            "query_variants": generate_query_variants(
                use_case["query_template"],
                count=5  # 每个用例生成 5 种不同的问法
            ),
            "expected_response_type": use_case["response_type"],
            "ground_truth": use_case.get("ground_truth", ""),
            "required_information": use_case.get("required_info", []),
            "test_category": use_case.get("test_type", "functional")
        }
        test_data.append(test_item)

    return test_data


def generate_query_variants(template, count=5):
    """
    生成问题的多种表达方式
    模拟真实用户的多样化提问
    """
    # 从模板生成变体
    variants = [
        f"我想{template}",
        f"请问怎么{template}？",
        f"t模板的详细步骤是什么？",
        f"帮忙{template}",
        f"关于{template}，能详细说说吗？",
        # ... 实际使用 LLM 生成更多变体
    ]
    return variants[:count]
```

## 第四步：实现自动化评估

```python
from ragas.metrics import faithfulness, answer_relevancy
from langchain_openai import ChatOpenAI
import pandas as pd

class AutomatedEvaluator:
    """
    自动化评估引擎
    """

    def __init__(self, llm_model):
        self.llm = llm_model
        self.metrics = {
            "faithfulness": faithfulness,
            "answer_relevancy": answer_relevancy,
        }

    def evaluate_single(self, query, response, context=None, ground_truth=None):
        """
        评估单条记录
        """
        from ragas import evaluate
        from datasets import Dataset

        data = {
            "user_input": [query],
            "response": [response],
            "retrieved_contexts": [context] if context else [[]],
            "ground_truth": [ground_truth] if ground_truth else [""]
        }

        dataset = Dataset.from_dict(data)

        # 只用无参考的指标（不需要 ground truth）
        available_metrics = []
        for name, metric in self.metrics.items():
            try:
                available_metrics.append(metric)
            except:
                pass

        if not available_metrics:
            # 如果 RAGAS 指标不可用，用 LLM 评估
            return self._llm_evaluate(query, response, ground_truth)

        result = evaluate(dataset, metrics=available_metrics)
        return result

    def _llm_evaluate(self, query, response, ground_truth=None):
        """
        用 LLM 作为评估器
        """
        eval_prompt = f"""
请评估以下客服回答的质量。

用户问题：{query}

客服回答：{response}
{f"参考正确答案：{ground_truth}" if ground_truth else ""}

请从以下维度打分（1-5分）：
1. 准确性：回答是否正确
2. 相关性：回答是否与问题相关
3. 完整性：回答是否完整
4. 清晰度：表达是否清晰

请用以下格式返回：
准确性: X
相关性: X
完整性: X
清晰度: X
"""
        result = self.llm.invoke(eval_prompt)
        return self._parse_eval_result(result.content)

    def _parse_eval_result(self, content):
        """解析评估结果"""
        scores = {}
        for line in content.split("\n"):
            if ":" in line:
                key, value = line.split(":")
                scores[key.strip()] = int(value.strip())
        return scores

    def evaluate_batch(self, test_dataset):
        """
        批量评估
        """
        results = []
        for item in test_dataset:
            # 使用变体列表中的第一个作为测试query
            query = item["query_variants"][0]
            response = self.get_ai_response(query)  # 调用你的 AI 系统

            eval_result = self.evaluate_single(
                query=query,
                response=response,
                ground_truth=item.get("ground_truth", "")
            )
            results.append({
                "id": item["id"],
                "category": item["category"],
                "intent": item["intent"],
                "eval": eval_result
            })

        return results

    def get_ai_response(self, query):
        """调用你的 AI 客服系统"""
        # 这里替换为你的 AI 系统调用
        return self.llm.invoke(query)
```

## 第五步：生成评估报告

```python
def generate_evaluation_report(results):
    """
    生成评估报告
    """
    df = pd.DataFrame(results)

    # 1. 整体得分
    all_scores = []
    for r in results:
        all_scores.append(r["eval"])
    avg_scores = {k: sum(d[k] for d in all_scores) / len(all_scores)
                  for k in all_scores[0].keys()}

    # 2. 按类别分析
    category_scores = df.groupby("category")["eval"].apply(
        lambda x: {k: sum(d[k] for d in x) / len(x) for k in x.iloc[0].keys()}
    )

    # 3. 低于阈值的问题
    low_score_threshold = 3.0
    low_score_items = []
    for r in results:
        eval_avg = sum(r["eval"].values()) / len(r["eval"])
        if eval_avg < low_score_threshold:
            low_score_items.append({
                "id": r["id"],
                "category": r["category"],
                "intent": r["intent"],
                "avg_score": eval_avg,
                "details": r["eval"]
            })

    # 4. 生成报告
    report = f"""
# AI 客服质量评估报告

## 整体得分
| 指标 | 平均分 | 目标分 | 达标情况 |
|------|--------|--------|---------|
| 准确性 | {avg_scores.get('准确性', 0):.2f} | 4.25 | {'✅' if avg_scores.get('准确性', 0) >= 4.25 else '❌'} |
| 相关性 | {avg_scores.get('相关性', 0):.2f} | 4.00 | {'✅' if avg_scores.get('相关性', 0) >= 4.00 else '❌'} |
| 完整性 | {avg_scores.get('完整性', 0):.2f} | 3.75 | {'✅' if avg_scores.get('完整性', 0) >= 3.75 else '❌'} |
| 清晰度 | {avg_scores.get('清晰度', 0):.2f} | 4.00 | {'✅' if avg_scores.get('清晰度', 0) >= 4.00 else '❌'} |

## 按类别分析
"""
    for cat, scores in category_scores.items():
        report += f"\n### {cat}\n"
        for k, v in scores.items():
            status = "✅" if v >= 3.5 else "⚠️" if v >= 3.0 else "❌"
            report += f"- {k}: {v:.2f} {status}\n"

    if low_score_items:
        report += f"\n## 需要优化的场景（得分 < {low_score_threshold}）\n"
        for item in low_score_items:
            report += f"- [{item['id']}] {item['category']} - {item['intent']}: {item['avg_score']:.2f}\n"

    return report


# 执行评估
evaluator = AutomatedEvaluator(llm_model=my_llm)
results = evaluator.evaluate_batch(test_dataset)
report = generate_evaluation_report(results)

print(report)
```

## 第六步：集成到 CI/CD

评估不是一次性活动，而是持续流程的一部分。

```yaml
# .github/workflows/ai-quality.yml
name: AI Quality Gate

on:
  push:
    branches: [main, develop]
  pull_request:
    branches: [main]

jobs:
  evaluate:
    runs-on: ubuntu-latest

    steps:
      - uses: actions/checkout@v4

      - name: Set up Python
        uses: actions/setup-python@v4
        with:
          python-version: '3.11'

      - name: Install dependencies
        run: |
          pip install ragas datasets langchain-openai pandas

      - name: Run Evaluation
        env:
          OPENAI_API_KEY: ${{ secrets.OPENAI_API_KEY }}
        run: |
          python -m evaluation.run_eval \
            --test-data data/eval_dataset.json \
            --output results/eval_results.json \
            --threshold 0.75

      - name: Publish Report
        uses: actions/upload-artifact@v3
        with:
          name: eval-report
          path: results/eval_results.json

      - name: Quality Gate
        run: |
          python -c "
          import json
          with open('results/eval_results.json') as f:
              results = json.load(f)

          # 检查关键指标
          if results['avg_accuracy'] < 0.85:
              print('❌ Accuracy below threshold')
              exit(1)
          if results['avg_safety'] < 0.99:
              print('❌ Safety below threshold')
              exit(1)
          print('✅ Quality gate passed')
          "
```

## 本节小结

构建企业级 AI 评估体系的七个步骤：

**1. 明确评估目标**：从业务痛点出发，定义需要回答的问题
**2. 设计指标体系**：业务指标 + 质量指标 + 效率指标，三层结构
**3. 构建测试数据集**：覆盖高频意图，生成多样化的问法变体
**4. 实现自动化评估**：用 LLM-as-Judge 或 RAGAS 框架自动化评分
**5. 生成评估报告**：可读性高的报告，识别问题和改进方向
**6. 集成到 CI/CD**：质量门槛自动化，每次发布前检查
**7. 持续监控与告警**：每日检查、每周报告，及时发现异常

**核心原则**：评估不是为了打分，而是为了驱动改进。评估结果必须转化为可执行的优化动作。
