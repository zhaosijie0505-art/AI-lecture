# 第六节 Autonomous Agent 架构

## 什么是 Autonomous Agent

前面的章节我们学习了 Agent 的各个核心组件：ReAct、Plan-and-Execute、Tool Use、Memory、Self-Reflection。现在把它们整合起来，构建一个真正能够**自主工作**的 Agent。

**Autonomous Agent** 的定义：
> 一个能够长期运行、自主设定目标、主动采取行动、从经验中学习、并在没有持续人类干预的情况下完成复杂任务的 AI 系统。

```
与传统 AI助手的区别：

传统 AI 助手：
用户提问 → AI 回答 → 结束（每次交互独立）

Autonomous Agent：
目标设定 → 自主规划 → 逐步执行 → 监控反馈 → 自我调整 → 持续工作
                                      ↑__________________|
                                            (学习循环)
```

## 完整实现

```python
import json
import re
from datetime import datetime
from typing import List, Dict, Any, Optional
from enum import Enum

class TaskState(Enum):
    PENDING = "pending"
    IN_PROGRESS = "in_progress"
    COMPLETED = "completed"
    FAILED = "failed"
    BLOCKED = "blocked"

class AutonomousAgent:
    """
    自主 Agent 完整实现
    """

    def __init__(
        self,
        llm,
        tools: Dict[str, callable],
        vector_store=None,
        embedding_model=None,
        config: dict = None
    ):
        self.llm = llm
        self.tools = tools
        self.config = config or self._default_config()

        # Memory 层
        self.sensory_memory = {}
        self.working_memory = {
            "tasks": [],           # 当前任务队列
            "completed": [],       # 已完成任务
            "failed": [],          # 失败任务
            "learned": []          # 学到的经验
        }
        self.longterm_memory = vector_store
        self.embedding_model = embedding_model

        # 状态
        self.current_goal = None
        self.execution_trace = []
        self.max_iterations = self.config.get("max_iterations", 50)

    def _default_config(self) -> dict:
        return {
            "max_iterations": 50,
            "reflection_enabled": True,
            "learning_enabled": True,
            "memory_compression_threshold": 20,
            "goal_confirmation_required": False,
        }

    def run(self, goal: str, confirmation_required: bool = None) -> dict:
        """
        自主运行主循环
        """
        self.current_goal = goal
        self.execution_trace = []
        confirm = confirmation_required if confirmation_required is not None \
                   else self.config["goal_confirmation_required"]

        # Step 1: 解析和确认目标
        parsed_goal = self._parse_goal(goal)
        if confirm:
            if not self._confirm_goal(parsed_goal):
                return {"status": "cancelled", "reason": "用户取消"}

        # Step 2: 规划执行步骤
        plan = self._create_plan(parsed_goal)
        self.execution_trace.append({
            "type": "plan",
            "content": plan,
            "timestamp": datetime.now().isoformat()
        })

        # Step 3: 执行计划
        result = self._execute_plan(plan)

        # Step 4: 学习与总结
        if self.config.get("learning_enabled"):
            self._learn_from_execution(result)

        return result

    def _parse_goal(self, goal: str) -> dict:
        """解析目标"""
        prompt = f"""
分析以下目标，提取关键信息：

目标：{goal}

请返回 JSON：
{{
  "objective": "核心目标",
  "constraints": ["约束1", ...],
  "success_criteria": ["成功标准1", ...],
  "estimated_steps": N,
  "potential_challenges": ["挑战1", ...]
}}
"""
        response = self.llm.generate(prompt)
        try:
            return json.loads(response)
        except:
            return {
                "objective": goal,
                "constraints": [],
                "success_criteria": ["任务完成"],
                "estimated_steps": 5
            }

    def _confirm_goal(self, parsed_goal: dict) -> bool:
        """确认目标（用户交互）"""
        print(f"📌 目标：{parsed_goal['objective']}")
        print(f"📋 步骤：约 {parsed_goal['estimated_steps']} 步")
        if parsed_goal['constraints']:
            print(f"⚠️ 约束：{', '.join(parsed_goal['constraints'])}")
        # 实际实现中，这里会调用 confirm 工具让用户确认
        return True

    def _create_plan(self, parsed_goal: dict) -> List[dict]:
        """创建执行计划"""
        prompt = f"""
为目标创建执行计划：

目标：{parsed_goal['objective']}
约束：{parsed_goal['constraints']}

请将目标分解成具体的执行步骤，每个步骤：
1. 有明确的输出
2. 可以通过工具或 LLM 生成完成
3. 按逻辑顺序排列

返回 JSON 数组：
[
  {{
    "step_id": 1,
    "description": "步骤描述",
    "tool": "使用的工具（如果有）",
    "params": {{"参数"}},
    "expected_output": "期望输出"
  }},
  ...
]
"""
        response = self.llm.generate(prompt)
        try:
            plan = json.loads(response)
            # 标记初始状态
            for step in plan:
                step["state"] = TaskState.PENDING
            return plan
        except:
            return [{"step_id": 1, "description": parsed_goal['objective'], "state": TaskState.PENDING}]

    def _execute_plan(self, plan: List[dict]) -> dict:
        """执行计划"""
        results = []
        all_successful = True

        for iteration in range(self.max_iterations):
            # 找到下一个待执行步骤
            next_step = self._get_next_step(plan, results)
            if not next_step:
                break  # 所有步骤完成或无法继续

            print(f"🔄 执行步骤 {next_step['step_id']}: {next_step['description']}")

            # 执行步骤
            step_result = self._execute_step(next_step)
            step_result["step_id"] = next_step["step_id"]
            results.append(step_result)

            # 记录到 trace
            self.execution_trace.append({
                "type": "step",
                "step": next_step["step_id"],
                "result": step_result,
                "timestamp": datetime.now().isoformat()
            })

            # 自我反思
            if self.config.get("reflection_enabled"):
                reflection = self._reflect_on_step(next_step, step_result)
                if reflection.get("needs_retry"):
                    print(f"⚠️ 步骤 {next_step['step_id']} 需要重试")
                    # 标记步骤重新执行
                    next_step["state"] = TaskState.PENDING

            # 检查是否完成
            if self._check_goal_completion(plan, results):
                print("✅ 目标完成")
                break

        return {
            "status": "completed" if all_successful else "partial",
            "goal": self.current_goal,
            "plan": plan,
            "results": results,
            "summary": self._summarize_results(results)
        }

    def _execute_step(self, step: dict) -> dict:
        """执行单个步骤"""
        tool = step.get("tool")
        params = step.get("params", {})

        if tool and tool in self.tools:
            try:
                # 使用工具执行
                result = self.tools[tool](**params)
                return {
                    "success": True,
                    "output": result
                }
            except Exception as e:
                return {
                    "success": False,
                    "error": str(e)
                }
        else:
            # 使用 LLM 直接生成
            output = self.llm.generate(step["description"])
            return {
                "success": True,
                "output": output
            }

    def _get_next_step(self, plan: List[dict], results: List[dict]) -> Optional[dict]:
        """获取下一个待执行的步骤"""
        completed_ids = {r["step_id"] for r in results if r.get("success")}

        for step in plan:
            if step["step_id"] not in completed_ids:
                return step

        return None

    def _reflect_on_step(self, step: dict, result: dict) -> dict:
        """反思单个步骤的执行"""
        prompt = f"""
反思以下步骤的执行：

步骤：{step['description']}
执行结果：{result}

请判断：
1. 步骤是否成功执行？
2. 输出是否符合预期？
3. 是否需要重试？

返回 JSON：
{{
  "success": true/false,
  "needs_retry": true/false,
  "reason": "判断理由",
  "improvement": "改进建议（如果有）"
}}
"""
        response = self.llm.generate(prompt)
        try:
            return json.loads(response)
        except:
            return {"needs_retry": False}

    def _check_goal_completion(self, plan: List[dict], results: List[dict]) -> bool:
        """检查目标是否完成"""
        if len(results) < len(plan):
            return False

        # 检查所有步骤是否成功
        return all(r.get("success") for r in results)

    def _summarize_results(self, results: List[dict]) -> str:
        """总结执行结果"""
        if not results:
            return "无结果"

        summary = f"共执行 {len(results)} 个步骤\n"
        successful = sum(1 for r in results if r.get("success"))
        summary += f"成功: {successful}/{len(results)}\n"

        return summary

    def _learn_from_execution(self, result: dict):
        """从执行结果中学习"""
        # 保存执行经验到长期记忆
        if self.longterm_memory:
            experience = {
                "goal": self.current_goal,
                "steps_count": len(result.get("results", [])),
                "success_rate": sum(1 for r in result.get("results", [])
                                   if r.get("success")) / max(len(result.get("results", [])), 1),
                "lessons": self._extract_lessons(result)
            }

            self.longterm_memory.add_memory(
                content=json.dumps(experience),
                memory_type="experience",
                metadata={"goal_type": "autonomous_task"}
            )
```

## 架构设计原则

### 1. 分层解耦

```
每层只关注自己的职责：
Goal Layer → "要做什么"
Planning Layer → "怎么做"
Execution Layer → "执行"
Memory Layer → "记住"
Reflection Layer → "检查和调整"
```

### 2. 模块化工具

```python
# 每个工具应该：
# 1. 单一职责
# 2. 有清晰的输入输出
# 3. 可以独立测试

class Tool(Protocol):
    name: str
    description: str

    def execute(self, **params) -> ToolResult
```

### 3. 错误容错

```python
# 不是每个错误都导致整个任务失败
def execute_with_fallback(agent, step, result):
    if not result.success:
        # 尝试替代方案
        alternative = agent._find_alternative_approach(step)
        if alternative:
            return agent._execute_step(alternative)
        else:
            # 记录失败，继续下一步（如果可行）
            step["state"] = TaskState.FAILED
            return result
```

### 4. 可观测性

```python
# 全程记录 trace，便于调试和审计
class ObservableAgent(AutonomousAgent):
    def __init__(self, *args, **kwargs):
        super().__init__(*args, **kwargs)
        self.events = EventEmitter()  # 发布订阅

    def _execute_step(self, step):
        result = super()._execute_step(step)
        self.events.emit("step_completed", {
            "step": step,
            "result": result,
            "trace": self.execution_trace
        })
        return result
```

## 本节小结

Autonomous Agent 架构的核心组件：

```
目标层：分解和确认高层目标
    ↓
规划层：Plan-and-Execute + ReAct
    ↓
执行层：Tool Use（工具调用）
    ↓
记忆层：Sensory + Working + Long-Term
    ↓
反思层：Self-Reflection + Error Correction
    ↓
感知层：自然语言理解
```

构建可靠 Autonomous Agent 的关键原则：
- **分层解耦**：每层职责单一
- **模块化工具**：工具可复用、可测试
- **错误容错**：局部失败不影响整体
- **可观测性**：全程记录 trace
