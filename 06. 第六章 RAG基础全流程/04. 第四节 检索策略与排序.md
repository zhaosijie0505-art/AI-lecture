# 第四节 检索策略与排序

## 检索的核心问题

检索的目标是：从海量文档中找到与用户问题最相关的 Top-K 个文档块。

```
检索质量 = 召回率 × 精确率

召回率：相关文档被检索到的比例
精确率：检索到的文档中真正相关的比例

RAG 目标：高召回率（不遗漏重要信息）× 高精确率（不引入无关信息）
```

## 检索方法对比

| 方法 | 原理 | 优点 | 缺点 |
|------|------|------|------|
| **关键词搜索** | BM25/TF-IDF | 精确、速度快 | 无法捕捉语义 |
| **向量搜索** | 语义相似度 | 理解语义 | 可能偏离字面 |
| **混合搜索** | 两者结合 | 兼顾精确和语义 | 实现复杂 |

### 1. 关键词搜索（BM25）

传统信息检索方法，基于词频统计：

```python
# 使用 BM25 进行关键词搜索
from rank_bm25 import BM25Okapi
import re

def bm25_search(query: str, documents: list, top_k: int = 5) -> list:
    """
    BM25 关键词搜索
    """
    # 分词
    tokenized_docs = [re.findall(r'\w+', doc.lower()) for doc in documents]
    tokenized_query = re.findall(r'\w+', query.lower())

    # 构建 BM25 索引
    bm25 = BM25Okapi(tokenized_docs)

    # 查询
    scores = bm25.get_scores(tokenized_query)

    # 返回 Top-K
    doc_scores = list(zip(documents, scores))
    doc_scores.sort(key=lambda x: x[1], reverse=True)

    return doc_scores[:top_k]

# 示例
docs = [
    "机器学习是人工智能的一个分支",
    "深度学习使用神经网络进行特征提取",
    "Python 是一种高级编程语言",
    "人工智能正在改变各行各业"
]

results = bm25_search("深度学习 神经网络", docs)
print(results)
# 返回：深度学习相关的文档排在前面
```

### 2. 向量搜索

基于语义相似度搜索：

```python
from sklearn.metrics.pairwise import cosine_similarity
import numpy as np

def vector_search(query: str, documents: list, embeddings, top_k: int = 5) -> list:
    """
    向量搜索
    """
    # 获取查询的 embedding
    query_embedding = get_embedding(query)

    # 获取所有文档的 embeddings
    doc_embeddings = [embed(doc) for doc in documents]

    # 计算相似度
    similarities = cosine_similarity([query_embedding], doc_embeddings)[0]

    # 排序返回 Top-K
    doc_scores = list(zip(documents, similarities))
    doc_scores.sort(key=lambda x: x[1], reverse=True)

    return doc_scores[:top_k]
```

### 3. 混合搜索（Hybrid Search）

结合关键词和向量搜索的优势：

```python
def hybrid_search(query: str, documents: list, top_k: int = 5, alpha: float = 0.5):
    """
    混合搜索：BM25 + 向量搜索

    alpha: 权重，0.5 表示两者同等重要
    alpha > 0.5: 更重视关键词匹配
    alpha < 0.5: 更重视语义相似
    """
    # 1. BM25 搜索
    bm25_results = bm25_search(query, documents, top_k=top_k * 2)
    bm25_scores = {doc: score for doc, score in bm25_results}
    max_bm25 = max(bm25_scores.values()) if bm25_scores else 1

    # 2. 向量搜索
    vector_results = vector_search(query, documents, top_k=top_k * 2)
    vector_scores = {doc: score for doc, score in vector_results}
    max_vector = max(vector_scores.values()) if vector_scores else 1

    # 3. 归一化并融合
    fused_scores = {}
    all_docs = set(bm25_scores.keys()) | set(vector_scores.keys())

    for doc in all_docs:
        bm25_norm = bm25_scores.get(doc, 0) / max_bm25
        vector_norm = vector_scores.get(doc, 0) / max_vector
        fused_scores[doc] = alpha * bm25_norm + (1 - alpha) * vector_norm

    # 4. 排序
    sorted_docs = sorted(fused_scores.items(), key=lambda x: x[1], reverse=True)
    return sorted_docs[:top_k]
```

## Reranking（重排序）

初步检索后，用更精细的模型重新排序，提升精确率。

### 1. Cross-Encoder Reranking

使用 Cross-Encoder 重新评估每个文档与查询的相关性：

```python
from sentence_transformers import CrossEncoder

# 加载 Cross-Encoder 模型
reranker = CrossEncoder('cross-encoder/ms-marco-MiniLM-L-6-v2')

def rerank_documents(query: str, documents: list, top_k: int = 5) -> list:
    """
    Cross-Encoder 重排序
    比 Bi-Encoder 更精确，但速度慢
    """
    # 构建 Query-Document 对
    pairs = [(query, doc) for doc in documents]

    # 获取相关性分数
    scores = reranker.predict(pairs)

    # 按分数排序
    doc_scores = list(zip(documents, scores))
    doc_scores.sort(key=lambda x: x[1], reverse=True)

    return doc_scores[:top_k]

# 示例
initial_results = vector_search("机器学习的应用", documents, k=20)
reranked = rerank_documents("机器学习的应用", initial_results, k=5)
```

### 2. Cohere Rerank（云服务）

```python
import cohere

cohere_client = cohere.Client("YOUR_API_KEY")

def cohere_rerank(query: str, documents: list, top_k: int = 5) -> list:
    """使用 Cohere Rerank API"""
    response = cohere_client.rerank(
        query=query,
        documents=documents,
        top_n=top_k,
        model="rerank-multilingual-v2.0"
    )

    return [(result.document.text, result.relevance_score)
            for result in response.results]
```

## 查询优化

### 1. Query Expansion（查询扩展）

将用户的一个问题扩展成多个角度的查询：

```python
def expand_query(query: str) -> list:
    """
    扩展查询，从多个角度检索
    """
    prompt = f"""
用户问题：{query}

请从不同角度生成 3 个等效的查询，用于检索更多相关内容：

要求：
- 保持原意
- 使用不同的表述方式
- 可以增加或减少细节

返回格式：
[
  "查询1",
  "查询2",
  "查询3"
]
"""
    expanded = llm.generate(prompt)
    queries = json.loads(expanded)
    queries.append(query)  # 保留原查询
    return queries

# 示例
# 原查询："公司年假政策"
# 扩展后：
# 1. "公司年假政策"
# 2. "年假有多少天"
# 3. "请假制度规定"
# 然后分别检索，再合并去重
```

### 2. Query Decomposition（查询分解）

将复杂问题分解成多个简单问题：

```python
def decompose_query(query: str) -> list:
    """
    分解复杂问题
    """
    prompt = f"""
用户问题：{query}

如果这个问题涉及多个子问题，请将其分解：

返回 JSON 数组：
[
  "子问题1",
  "子问题2"
]

如果没有子问题，返回空数组。
"""
    result = llm.generate(prompt)
    sub_queries = json.loads(result)
    return sub_queries if sub_queries else [query]

# 示例
# 原查询："特斯拉的市值和比亚迪比怎么样？"
# 分解后：
# 1. "特斯拉市值"
# 2. "比亚迪市值"
# 分别检索后合并回答
```

### 3. HyDE（Hypothetical Document Embeddings）

用 LLM 生成一个"假设答案"，再用这个答案去检索：

```python
defhyde_search(query: str, vector_store, llm) -> list:
    """
    HyDE 搜索：用假设答案引导检索
    """
    # Step 1: 让 LLM 生成一个假设的回答
    hypothetical_prompt = f"""
请针对以下问题，生成一个简短的回答。
这个回答可以是假设的，不需要完全准确。

问题：{query}

回答："""

    hypothetical_answer = llm.generate(hypothetical_prompt)

    # Step 2: 用这个假设答案去检索相关文档
    # 因为假设答案可能包含正确的关键词和概念
    results = vector_store.similarity_search(hypothetical_answer, k=5)

    return results
```

## 检索参数调优

```python
# LangChain 检索参数
retriever = vector_store.as_retriever(
    search_type="similarity",  # 或 "mmr"
    search_kwargs={
        "k": 5,                    # 返回 5 个文档
        "score_threshold": 0.8,    # 相似度阈值（可选）
        "fetch_k": 20,             # MMR 模式：先取 20 个
        "lambda_mult": 0.5,        # MMR：相关性和多样性平衡
    }
)

# MMR（最大边际相关）实现
def mmr_search(query: str, documents: list, embeddings, k: int = 5, fetch_k: int = 20, lambda_mult: float = 0.5):
    """
    MMR：平衡相关性和多样性
    lambda_mult 越低，多样性越高
    """
    # 获取查询 embedding
    query_embedding = get_embedding(query)

    # 获取所有文档的 embeddings
    doc_embeddings = [get_embedding(doc) for doc in documents]

    # 计算每个文档与查询的相似度
    similarities = cosine_similarity([query_embedding], doc_embeddings)[0]

    # MMR 选择
    selected = []
    remaining = list(range(len(documents)))

    for _ in range(k):
        best_score = -float('inf')
        best_idx = None

        for idx in remaining:
            # 相关性分数
            relevance = similarities[idx]
            # 多样性分数（与已选文档的最大相似度）
            if selected:
                selected_embeddings = [doc_embeddings[i] for i in selected]
                max_similarity_to_selected = max(
                    cosine_similarity([doc_embeddings[idx]], selected_embeddings)[0]
                )
            else:
                max_similarity_to_selected = 0

            # MMR 公式
            mmr_score = lambda_mult * relevance - (1 - lambda_mult) * max_similarity_to_selected

            if mmr_score > best_score:
                best_score = mmr_score
                best_idx = idx

        selected.append(best_idx)
        remaining.remove(best_idx)

    return [(documents[i], similarities[i]) for i in selected]
```

## 本节小结

**三种检索方法**：关键词搜索（BM25）、向量搜索（语义相似度）、混合搜索（两者结合）。

**Reranking**：用 Cross-Encoder 或 Cohere 在初步检索后重新排序，提升精确率。

**查询优化**：Query Expansion（多角度扩展）、Query Decomposition（分解复杂问题）、HyDE（假设答案引导检索）。

**MMR**：最大边际相关，在相关性和多样性之间取得平衡，适合返回多个文档的场景。
