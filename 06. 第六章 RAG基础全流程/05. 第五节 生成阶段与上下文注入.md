# 第五节 生成阶段与上下文注入

## 生成阶段的关键

检索找到了相关文档，但最终效果还取决于**如何把文档组织成 Prompt 发给 LLM**。

同样的检索结果，不同的 Prompt 写法，生成质量可能相差很大。

```
好的上下文组织：
- 清晰标识文档来源
- 相关内容放在前面
- 避免信息过载

差的上下文组织：
- 所有文档混在一起
- 无关内容太多
- 上下文超出 LLM 限制
```

### 带引用标注的 Prompt

```python
# 让回答包含引用标注
citation_template = """
你是一个 AI 助手。请根据以下参考资料回答用户问题。

参考资料：
{context}

用户问题：{question}

回答要求：
1. 基于给定的参考资料回答
2. 在回答中用 [来源N] 标注信息来源
3. 如果参考资料中没有相关信息，直接说你不知道

回答：
"""

# 上下文格式（每个文档带编号）：
# [来源1] 文档内容...
# [来源2] 文档内容...
# [来源3] 文档内容...

def format_context_with_citations(docs: list, max_chars: int = 4000) -> str:
    """格式化带编号的上下文"""
    formatted = []
    total_chars = 0

    for i, doc in enumerate(docs, 1):
        # 截断过长的单个文档
        content = doc.page_content if hasattr(doc, 'page_content') else str(doc)
        if len(content) > 500:
            content = content[:500] + "..."

        # 检查总长度
        if total_chars + len(content) > max_chars:
            break

        formatted.append(f"[来源{i}] {content}")
        total_chars += len(content)

    return "\n\n".join(formatted)
```

def answer_with_reasoning(chain, question: str, docs: list) -> str:

## 上下文组织策略

### 1. 按相关度排序

把最相关的文档放在前面：

```python
def organize_context_by_relevance(docs_with_scores: list) -> str:
    """
    按相关度排序上下文
    docs_with_scores: [(doc, similarity_score), ...]
    """
    # 按相似度降序排列
    sorted_docs = sorted(docs_with_scores, key=lambda x: x[1], reverse=True)

    context_parts = []
    for i, (doc, score) in enumerate(sorted_docs, 1):
        content = doc.page_content if hasattr(doc, 'page_content') else str(doc)
        context_parts.append(f"[文档{i}](相似度: {score:.2f})\n{content}")

    return "\n\n".join(context_parts)
```

### 2. 上下文压缩

移除上下文中无关的冗余内容：

```python
def compress_context(docs: list, query: str, max_tokens: int = 4000) -> str:
    """
    压缩上下文，保留与问题相关的部分
    """
    compressed_parts = []
    total_tokens = 0

    for doc in docs:
        content = doc.page_content if hasattr(doc, 'page_content') else str(doc)

        # 用 LLM 提取与问题相关的部分
        extract_prompt = f"""
从以下文档中提取与问题相关的内容。

问题：{query}

文档：
{content}

只返回与问题直接相关的内容，不要添加解释。如果文档与问题无关，返回空字符串。
"""
        relevant_content = llm.generate(extract_prompt).strip()

        if relevant_content:
            # 检查 token 限制
            content_tokens = count_tokens(relevant_content)
            if total_tokens + content_tokens > max_tokens:
                break

            compressed_parts.append(relevant_content)
            total_tokens += content_tokens

    return "\n\n".join(compressed_parts)
```

### 3. 上下文窗口滑动

处理长文档时，使用滑动窗口：

```python
class SlidingContextWindow:
    """滑动上下文窗口"""

    def __init__(self, window_size: int = 3000, overlap: int = 500):
        self.window_size = window_size
        self.overlap = overlap

    def get_context_window(self, doc: str, query: str) -> str:
        """
        获取与查询最相关的上下文窗口
        """
        # 找到查询关键词在文档中的位置
        query_keywords = query.split()
        positions = []

        for keyword in query_keywords:
            pos = doc.lower().find(keyword.lower())
            if pos >= 0:
                positions.append(pos)

        if not positions:
            return doc[:self.window_size]

        # 以关键词位置为中心提取窗口
        center = sum(positions) // len(positions)
        start = max(0, center - self.window_size // 2)
        end = min(len(doc), center + self.window_size // 2)

        window = doc[start:end]

        # 添加边界上下文（overlap）
        if start > 0:
            window = "... " + doc[start:start + self.overlap] + window
        if end < len(doc):
            window = window + doc[end - self.overlap:end] + " ..."

        return window
```

## 生成策略

### 1. 单次生成（Standard Generation）

最常用的方式：

```python
def standard_generate(question: str, docs: list, llm) -> str:
    """标准单次生成"""
    context = format_context(docs)

    prompt = f"""
基于以下参考资料回答问题。回答要准确、完整。

参考资料：
{context}

问题：{question}

回答："""

    return llm.generate(prompt)
```

### 2. 逐步生成（Step-by-Step Generation）

复杂问题分步骤回答：

```python
def step_by_step_generate(question: str, docs: list, llm) -> str:
    """分步骤生成答案"""

    # Step 1: 识别问题涉及的关键点
    identify_prompt = f"""
分析以下问题涉及哪些关键点或方面：

问题：{question}

列出关键点，每个一行：
"""
    key_points = llm.generate(identify_prompt)

    # Step 2: 逐个关键点从参考资料中获取信息
    answers = []
    for point in key_points.strip().split('\n'):
        if not point.strip():
            continue

        point_prompt = f"""
参考资料：
{format_context(docs)}

关于这个方面：{point}
请给出简要回答：
"""
        answer = llm.generate(point_prompt)
        answers.append(f"**{point}**：{answer}")

    # Step 3: 综合成完整答案
    synthesis_prompt = f"""
原问题：{question}

各部分回答：
{' '.join(answers)}

请将以上内容综合成一个连贯的完整答案：
"""

    return llm.generate(synthesis_prompt)
```

### 3. 生成后验证

生成答案后，检查是否基于参考资料：

```python
def generate_with_verification(question: str, docs: list, llm) -> tuple:
    """生成答案并验证"""

    # Step 1: 生成答案
    answer = standard_generate(question, docs, llm)

    # Step 2: 验证答案
    verify_prompt = f"""
请验证以下回答是否基于给定的参考资料。

原问题：{question}

参考资料：
{format_context(docs)}

回答：
{answer}

验证要求：
1. 回答中的每个声明是否能在参考资料中找到依据？
2. 是否有参考资料中没有的内容（可能是幻觉）？

返回 JSON：
{{
  "is_faithful": true/false,
  "unsupported_claims": ["不支持的声明1", ...],
  "verification_notes": "验证说明"
}}
"""
    verification = json.loads(llm.generate(verify_prompt))

    if not verification["is_faithful"]:
        # 如果有幻觉，重新生成
        refine_prompt = f"""
请重新回答问题，注意：
1. 只基于参考资料中的信息
2. 不要添加参考资料中没有的内容

参考资料：
{format_context(docs)}

问题：{question}

回答：
"""
        answer = llm.generate(refine_prompt)

    return answer, verification
```

### 2. 控制回答长度

```python
# 根据问题类型调整回答长度
def get_length_instruction(question: str) -> str:
    """根据问题类型返回回答长度要求"""

    if any(kw in question for kw in ["多少", "几个", "列表"]):
        return "请简洁列出要点"
    elif any(kw in question for kw in ["解释", "说明", "为什么"]):
        return "请详细解释原因"
    elif any(kw in question for kw in ["比较", "对比"]):
        return "请对比分析，列出异同点"
    else:
        return "请简洁、准确回答"
```

### 3. 多次采样一致性

```python
def consensus_generate(question: str, docs: list, llm, n_samples: int = 3) -> str:
    """多次采样，取一致性最高的答案"""

    answers = []
    for _ in range(n_samples):
        answer = standard_generate(question, docs, llm)
        answers.append(answer)

    # 让 LLM 判断哪个答案最准确，或综合成最佳答案
    consensus_prompt = f"""
问题：{question}

以下是对这个问题的 {n_samples} 个回答：
{' '.join([f'回答{i+1}：{a}' for i, a in enumerate(answers)])}

请判断哪个回答最准确、最完整，或者综合成一个最佳答案。
只返回最终答案，不要解释判断过程。
"""

    return llm.generate(consensus_prompt)
```

## 本节小结

生成阶段的关键点：

**Prompt 模板**：基础模板、带引用标注的模板、带推理引导的模板。

**上下文组织**：按相关度排序、上下文压缩（移除无关内容）、滑动窗口处理长文档。

**生成策略**：单次生成（标准）、分步骤生成（复杂问题）、生成后验证（防止幻觉）。

**优化技巧**：指定输出格式、控制回答长度、多次采样取一致性。
