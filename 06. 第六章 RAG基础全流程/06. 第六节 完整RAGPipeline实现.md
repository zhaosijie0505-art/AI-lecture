import os

from dataclasses import dataclass
from abc import ABC, abstractmethod
import re

# ============================================================
# 第一部分：文档处理
# ============================================================

@dataclass
class Document:
    """文档数据结构"""
    page_content: str
    metadata: Dict[str, Any]

    def __repr__(self):
        return f"Document(content_len={len(self.page_content)}, meta={self.metadata})"

class DocumentLoader(ABC):
    """文档加载器基类"""
    @abstractmethod
    def load(self, path: str) -> List[Document]:
        pass

class TextFileLoader(DocumentLoader):
    """文本文件加载器"""
    def load(self, path: str) -> List[Document]:
        with open(path, 'r', encoding='utf-8') as f:
            content = f.read()
        return [Document(page_content=content, metadata={"source": path})]

class MarkdownLoader(DocumentLoader):
    """Markdown 文件加载器"""
    def load(self, path: str) -> List[Document]:
        with open(path, 'r', encoding='utf-8') as f:
            content = f.read()

        # 按标题分割成多个文档
        sections = re.split(r'(?=^#{1,6}\s+)', content, flags=re.MULTILINE)

        docs = []
        for i, section in enumerate(sections):
            if section.strip():
                docs.append(Document(
                    page_content=section.strip(),
                    metadata={"source": path, "section": i}
                ))

        return docs

class PDFLoader(DocumentLoader):
    """PDF 文件加载器（简化版，实际用 pypdf）"""
    def load(self, path: str) -> List[Document]:
        # 这里简化处理，实际应使用 pypdf 或 pdfplumber
        with open(path, 'rb') as f:
            # 伪代码：实际使用 PDF 解析库
            content = f.read().decode('utf-8', errors='ignore')

        return [Document(page_content=content, metadata={"source": path})]

# 第二部分：文本分块
# ============================================================

class TextSplitter:
    """文本分块器"""

    def __init__(
        self,
        chunk_size: int = 1000,
        chunk_overlap: int = 200,
        separators: List[str] = None
    ):
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.separators = separators or ["\n\n", "\n", ". ", " "]

    def split_text(self, text: str) -> List[str]:
        """将文本分割成块"""
        if len(text) <= self.chunk_size:
            return [text]

        chunks = []
        start = 0

        while start < len(text):
            end = start + self.chunk_size

            # 尝试在分隔符处分割
            for sep in self.separators:
                split_point = text.rfind(sep, start, end)
                if split_point > start:
                    end = split_point + len(sep)
                    break

            chunk = text[start:end].strip()
            if chunk:
                chunks.append(chunk)

            # 滑动窗口
            start = end - self.chunk_overlap

        return chunks

    def split_documents(self, documents: List[Document]) -> List[Document]:
        """分割文档列表"""
        chunks = []

        for doc in documents:
            text_chunks = self.split_text(doc.page_content)
            for i, chunk in enumerate(text_chunks):
                chunks.append(Document(
                    page_content=chunk,
                    metadata={**doc.metadata, "chunk": i}
                ))

        return chunks

# 第三部分：Embedding
# ============================================================

class EmbeddingModel(ABC):
    """Embedding 模型基类"""
    @abstractmethod
    def embed(self, texts: List[str]) -> List[List[float]]:
        pass

class OpenAIEmbedding(EmbeddingModel):
    """OpenAI Embedding"""

    def __init__(self, model: str = "text-embedding-3-small"):
        self.model = model
        self.client = None  # 延迟初始化

    def _get_client(self):
        if self.client is None:
            from openai import OpenAI
            self.client = OpenAI()
        return self.client

    def embed(self, texts: List[str]) -> List[List[float]]:
        """生成文本的 embedding 向量"""
        client = self._get_client()

        response = client.embeddings.create(
            model=self.model,
            input=texts
        )

        return [item.embedding for item in response.data]

class LocalEmbedding(EmbeddingModel):
    """本地 Embedding 模型（使用 Sentence Transformers）"""

    def __init__(self, model_name: str = "BAAI/bge-large-zh-v1.5"):
        from sentence_transformers import SentenceTransformer
        self.model = SentenceTransformer(model_name)

    def embed(self, texts: List[str]) -> List[List[float]]:
        """生成文本的 embedding 向量"""
        embeddings = self.model.encode(texts, normalize=True)
        return embeddings.tolist()

# ============================================================
# 第四部分：向量存储
# ============================================================

class VectorStore(ABC):
    """向量存储基类"""
    @abstractmethod
    def add_documents(self, documents: List[Document], embeddings: List[List[float]]):
        pass

    @abstractmethod
    def similarity_search(
        self,
        query_embedding: List[float],
        k: int = 5
    ) -> List[tuple]:
        pass

class SimpleVectorStore(VectorStore):
    """简单向量存储（内存版，用于演示）"""

    def __init__(self):
        self.documents: List[Document] = []
        self.embeddings: List[List[float]] = []

    def add_documents(self, documents: List[Document], embeddings: List[List[float]]):
        self.documents.extend(documents)
        self.embeddings.extend(embeddings)

    def similarity_search(
        self,
        query_embedding: List[float],
        k: int = 5
    ) -> List[tuple]:
        """计算相似度并返回 Top-K"""
        from numpy import dot
        from numpy.linalg import norm

        similarities = []
        for doc, embedding in zip(self.documents, self.embeddings):
            # 余弦相似度
            sim = dot(query_embedding, embedding) / (norm(query_embedding) * norm(embedding))
            similarities.append((doc, sim))

        # 排序并返回 Top-K
        similarities.sort(key=lambda x: x[1], reverse=True)
        return similarities[:k]

class ChromaVectorStore(VectorStore):
    """ChromaDB 向量存储"""

    def __init__(self, persist_directory: str = "./chroma_db"):
        import chromadb
        self.client = chromadb.PersistentClient(path=persist_directory)
        self.collection = None

    def create_collection(self, name: str = "documents"):
        self.collection = self.client.get_or_create_collection(name=name)

    def add_documents(self, documents: List[Document], embeddings: List[List[float]]):
        if self.collection is None:
            self.create_collection()

        ids = [f"doc_{i}" for i in range(len(documents))]
        texts = [doc.page_content for doc in documents]
        metadatas = [doc.metadata for doc in documents]

        self.collection.add(
            ids=ids,
            embeddings=embeddings,
            documents=texts,
            metadatas=metadatas
        )

    def similarity_search(
        self,
        query_embedding: List[float],
        k: int = 5
    ) -> List[tuple]:
        results = self.collection.query(
            query_embeddings=[query_embedding],
            n_results=k
        )

        docs = []
        for i in range(len(results['documents'][0])):
            doc = Document(
                page_content=results['documents'][0][i],
                metadata=results['metadatas'][0][i]
            )
            score = results['distances'][0][i]
            docs.append((doc, 1 - score))  # Chroma 距离转相似度

        return docs

# ============================================================
# 第五部分：检索器
# ============================================================

class BaseRetriever(ABC):
    """检索器基类"""
    @abstractmethod
    def get_relevant_documents(self, query: str) -> List[tuple]:
        pass

class VectorRetriever(BaseRetriever):
    """向量检索器"""

    def __init__(
        self,
        vector_store: VectorStore,
        embedding_model: EmbeddingModel,
        k: int = 5
    ):
        self.vector_store = vector_store
        self.embedding_model = embedding_model
        self.k = k

    def get_relevant_documents(self, query: str) -> List[tuple]:
        query_embedding = self.embedding_model.embed([query])[0]
        return self.vector_store.similarity_search(query_embedding, k=self.k)

class KeywordRetriever(BaseRetriever):
    """关键词检索器（BM25）"""
    # 简化实现，可替换为 rank_bm25

    def __init__(self, documents: List[Document], k: int = 5):
        self.documents = documents
        self.k = k

    def get_relevant_documents(self, query: str) -> List[tuple]:
        query_terms = set(query.lower().split())

        scored = []
        for doc in self.documents:
            doc_terms = set(doc.page_content.lower().split())
            overlap = len(query_terms & doc_terms)
            if overlap > 0:
                score = overlap / len(query_terms)
                scored.append((doc, score))

        scored.sort(key=lambda x: x[1], reverse=True)
        return scored[:self.k]

class HybridRetriever(BaseRetriever):
    """混合检索器"""

    def __init__(
        self,
        vector_store: VectorStore,
        documents: List[Document],
        embedding_model: EmbeddingModel,
        k: int = 5,
        alpha: float = 0.5
    ):
        self.vector_retriever = VectorRetriever(vector_store, embedding_model, k * 2)
        self.keyword_retriever = KeywordRetriever(documents, k * 2)
        self.k = k
        self.alpha = alpha

    def get_relevant_documents(self, query: str) -> List[tuple]:
        # 向量检索
        vector_results = self.vector_retriever.get_relevant_documents(query)
        vector_scores = {doc.page_content: score for doc, score in vector_results}

        # 关键词检索
        keyword_results = self.keyword_retriever.get_relevant_documents(query)
        keyword_scores = {doc.page_content: score for doc, score in keyword_results}

        # 融合
        all_docs = set(vector_scores.keys()) | set(keyword_scores.keys())
        fused_scores = []

        for content in all_docs:
            v_score = vector_scores.get(content, 0)
            k_score = keyword_scores.get(content, 0)

            max_v = max(vector_scores.values()) if vector_scores else 1
            max_k = max(keyword_scores.values()) if keyword_scores else 1

            fused_score = self.alpha * (v_score / max_v) + (1 - self.alpha) * (k_score / max_k)

            # 找到对应的 Document 对象
            doc = next(d for d in self.keyword_retriever.documents if d.page_content == content)
            fused_scores.append((doc, fused_score))

        fused_scores.sort(key=lambda x: x[1], reverse=True)
        return fused_scores[:self.k]

# 第六部分：RAG 生成器
# ============================================================

class RAGGenerator:
    """RAG 生成器"""

    def __init__(
        self,
        retriever: BaseRetriever,
        llm: Any,
        prompt_template: str = None
    ):
        self.retriever = retriever
        self.llm = llm
        self.prompt_template = prompt_template or DEFAULT_PROMPT_TEMPLATE

    def generate(self, query: str, return_docs: bool = False) -> Dict[str, Any]:
        """
        生成回答

        Args:
            query: 用户问题
            return_docs: 是否返回检索到的文档

        Returns:
            {"answer": str, "documents": [(doc, score), ...]}
        """
        # 1. 检索相关文档
        relevant_docs = self.retriever.get_relevant_documents(query)

        # 2. 构建上下文
        context = self._build_context(relevant_docs)

        # 3. 生成回答
        prompt = self.prompt_template.format(context=context, question=query)
        answer = self.llm.generate(prompt)

        result = {"answer": answer}
        if return_docs:
            result["documents"] = relevant_docs

        return result

    def _build_context(self, docs_with_scores: List[tuple]) -> str:
        """构建上下文"""
        context_parts = []
        for i, (doc, score) in enumerate(docs_with_scores, 1):
            context_parts.append(
                f"[来源{i}](相似度: {score:.2f})\n{doc.page_content}"
            )
        return "\n\n".join(context_parts)

DEFAULT_PROMPT_TEMPLATE = """你是一个 AI 助手。请根据以下参考资料回答用户问题。

{context}

用户问题：{question}

回答要求：
1. 基于给定的参考资料回答
2. 如果参考资料中没有相关信息，直接说你不知道
3. 在回答中用 [来源N] 标注信息来源
4. 回答要清晰、有条理

回答："""

# 第七部分：完整 RAG Pipeline
# ============================================================

class RAGPipeline:
    """完整的 RAG Pipeline"""

    def __init__(
        self,
        embedding_model: EmbeddingModel,
        vector_store: VectorStore,
        llm: Any,
        chunk_size: int = 1000,
        chunk_overlap: int = 200,
        retrieval_k: int = 5,
        retrieval_type: str = "vector"  # "vector", "keyword", "hybrid"
    ):
        self.embedding_model = embedding_model
        self.vector_store = vector_store
        self.llm = llm
        self.chunk_size = chunk_size
        self.chunk_overlap = chunk_overlap
        self.retrieval_type = retrieval_type
        self.retrieval_k = retrieval_k

        self.splitter = TextSplitter(chunk_size, chunk_overlap)
        self.documents: List[Document] = []

    def ingest(self, file_path: str, file_type: str = "auto"):
        """
        入库文档

        Args:
            file_path: 文件路径
            file_type: 文件类型，支持 "txt", "md", "pdf", "auto"(自动检测)
        """
        # 1. 加载文档
        if file_type == "auto":
            if file_path.endswith('.md'):
                file_type = "md"
            elif file_path.endswith('.pdf'):
                file_type = "pdf"
            else:
                file_type = "txt"

        if file_type == "md":
            loader = MarkdownLoader()
        elif file_type == "pdf":
            loader = PDFLoader()
        else:
            loader = TextFileLoader()

        documents = loader.load(file_path)
        print(f"加载了 {len(documents)} 个文档")

        # 2. 分块
        chunks = self.splitter.split_documents(documents)
        print(f"分块后得到 {len(chunks)} 个块")

        # 3. 生成 Embedding
        texts = [chunk.page_content for chunk in chunks]
        embeddings = self.embedding_model.embed(texts)
        print(f"生成了 {len(embeddings)} 个 embedding")

        # 4. 存入向量数据库
        self.vector_store.add_documents(chunks, embeddings)
        self.documents.extend(chunks)
        print(f"成功入库 {len(chunks)} 个文档块")

    def query(self, question: str, return_docs: bool = False) -> Dict[str, Any]:
        """
        查询

        Args:
            question: 用户问题
            return_docs: 是否返回检索到的文档

        Returns:
            {"answer": str, "documents": [(doc, score), ...]}
        """
        # 创建检索器
        if self.retrieval_type == "vector":
            retriever = VectorRetriever(self.vector_store, self.embedding_model, self.retrieval_k)
        elif self.retrieval_type == "keyword":
            retriever = KeywordRetriever(self.documents, self.retrieval_k)
        else:  # hybrid
            retriever = HybridRetriever(
                self.vector_store, self.documents,
                self.embedding_model, self.retrieval_k
            )

        # 创建生成器
        generator = RAGGenerator(retriever, self.llm)

        # 生成回答
        return generator.generate(question, return_docs=return_docs)

# ============================================================

def main():
    """使用示例"""
    from openai import OpenAI

    # 配置
    openai_client = OpenAI()

    # 初始化组件
    embedding_model = OpenAIEmbedding(model="text-embedding-3-small")
    vector_store = SimpleVectorStore()  # 生产环境用 ChromaVectorStore
    llm = openai_client  # 直接传 client

    # 创建 Pipeline
    rag = RAGPipeline(
        embedding_model=embedding_model,
        vector_store=vector_store,
        llm=llm,
        chunk_size=1000,
        chunk_overlap=200,
        retrieval_k=5,
        retrieval_type="hybrid"
    )

    # 入库文档
    rag.ingest("documents/faq.md", file_type="md")
    rag.ingest("documents/manual.txt", file_type="txt")

    # 查询
    result = rag.query("你们公司的年假政策是什么？", return_docs=True)

    print("\n回答：")
    print(result["answer"])

    print("\n使用的参考资料：")
    for doc, score in result["documents"]:
        print(f"  - [{score:.2f}] {doc.page_content[:100]}...")

if __name__ == "__main__":
    main()

def batch_ingest(pipeline, documents: List[str]):
    """批量入库"""
    batch_size = 100
    for i in range(0, len(documents), batch_size):
        batch = documents[i:i+batch_size]
        pipeline.ingest_batch(batch)
        print(f"已处理 {min(i+batch_size, len(documents))}/{len(documents)}")

# 2. 缓存 Embedding（避免重复计算）
class CachedEmbeddingModel(EmbeddingModel):
    def __init__(self, base_model: EmbeddingModel):
        self.base_model = base_model
        self.cache = {}

    def embed(self, texts: List[str]) -> List[List[float]]:
        results = []
        uncached = []

        for i, text in enumerate(texts):
            if text in self.cache:
                results.append(self.cache[text])
            else:
                results.append(None)
                uncached.append((i, text))

        if uncached:
            new_embeddings = self.base_model.embed([t for _, t in uncached])
            for (i, text), embedding in zip(uncached, new_embeddings):
                results[i] = embedding
                self.cache[text] = embedding

        return results

## 本节小结

完整 RAG Pipeline 的核心组件：

**文档处理**：加载器（支持 txt/md/pdf）→ 分块器（RecursiveCharacter）→ Document 数据结构

**Embedding**：OpenAI Embedding 或本地模型（BGE）

**向量存储**：SimpleVectorStore（测试）、ChromaDB（开发）、Milvus/Pinecone（生产）

**检索器**：VectorRetriever（语义）、KeywordRetriever（BM25）、HybridRetriever（混合）

**生成器**：RAGGenerator（Prompt 模板 + LLM）

**Pipeline**：ingest（入库）+ query（查询）两阶段

生产环境建议：选择合适的向量数据库、添加缓存、优化批量处理、建立监控指标。
