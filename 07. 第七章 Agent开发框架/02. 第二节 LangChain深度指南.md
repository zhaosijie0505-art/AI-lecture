## LCEL（LangChain Expression Language）

LCEL 是 LangChain 的声明式 Chain 定义方式：

```python
# LCEL 使用 | 符号连接组件
chain = prompt | model | output_parser

# 等价于：
chain = LLMChain(prompt=prompt, llm=model, output_parser=output_parser)
```

### 基础用法

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import PromptTemplate
from langchain.schema import StrOutputParser

# 简单翻译 Chain
llm = ChatOpenAI(model="gpt-4o")

prompt = PromptTemplate.from_template(
    "把以下中文翻译成英文：{text}"
)

chain = prompt | llm | StrOutputParser()

result = chain.invoke({"text": "你好，世界"})
print(result)  # Hello, world
```

### 复杂 Chain

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import PromptTemplate, ChatPromptTemplate
from langchain.schema import StrOutputParser
from langchain.schema import HumanMessage, SystemMessage

# 带系统消息的 Chain
system_prompt = "你是一个专业的翻译员，只翻译，不添加任何解释。"

chat_prompt = ChatPromptTemplate.from_messages([
    SystemMessage(content=system_prompt),
    HumanMessage(content="把以下{Source_lang}翻译成{_target_lang}：{text}")
])

# 组合 Chain
translation_chain = chat_prompt | llm | StrOutputParser()

result = translation_chain.invoke({
    "Source_lang": "中文",
    "target_lang": "英文",
    "text": "机器学习是人工智能的一个分支"
})

print(result)  # Machine learning is a branch of artificial intelligence.
```

### 并行执行

```python
from langchain_core.runnables import parallel

# 并行执行多个任务
llm = ChatOpenAI(model="gpt-4o")

prompt = PromptTemplate.from_template("简述{topic}的核心概念")

# 并行执行多个翻译
chain = prompt | llm | {"en": StrOutputParser()}

# 用 parallel 同时执行多个任务
def translate_to_lang(lang: str):
    return PromptTemplate.from_template(
        f"翻译成{{text}}"
    ) | llm | StrOutputParser()

results = parallel(
    lambda: translate_to_lang("英文").invoke({"text": "机器学习"}),
    lambda: translate_to_lang("日文").invoke({"text": "机器学习"}),
    lambda: translate_to_lang("法文").invoke({"text": "机器学习"})
)

# results: ["Machine learning", "機械学習", "Apprentissage automatique"]
```

## Tool 与 ToolCall

### 定义 Tool

```python
from langchain.tools import tool, StructuredTool
from pydantic import BaseModel

# 简单 Tool
@tool
def get_weather(city: str) -> str:
    """获取城市天气

    Args:
        city: 城市名称
    """
    # 实际调用天气 API
    return f"{city}今天晴天，25度"

# 带参数的 Tool（使用 Pydantic 模型）
class WeatherInput(BaseModel):
    city: str
    unit: str = "celsius"  # 可选参数，默认 celsius

@tool
def get_weather_structured(args: WeatherInput) -> str:
    """获取城市天气（结构化参数版）"""
    temp = 25 if args.unit == "celsius" else 77
    unit_str = "°C" if args.unit == "celsius" else "°F"
    return f"{args.city}今天晴天，{temp}{unit_str}"

# StructuredTool 方式
def get_stock_price(symbol: str, date: str) -> dict:
    """获取股票价格"""
    return {"symbol": symbol, "price": 150.5, "date": date}

stock_tool = StructuredTool.from_function(
    func=get_stock_price,
    name="get_stock_price",
    description="获取股票的历史价格或当前价格",
    args_schema={
        "type": "object",
        "properties": {
            "symbol": {"type": "string", "description": "股票代码，如 AAPL"},
            "date": {"type": "string", "description": "日期，格式 YYYY-MM-DD"}
        },
        "required": ["symbol"]
    }
)
```

### Tool 使用实战

```python
from langchain.agents import AgentExecutor, create_openai_functions_agent
from langchain_openai import ChatOpenAI

llm = ChatOpenAI(model="gpt-4o", temperature=0)

# 定义工具列表
tools = [get_weather, stock_tool]

# 创建 Agent
prompt = ChatPromptTemplate.from_messages([
    ("system", "你是一个有用的AI助手。"),
    ("human", "{input}"),
    ("placeholder", "{agent_scratchpad}")  # Agent 的内部工作区
])

agent = create_openai_functions_agent(llm, tools, prompt)
agent_executor = AgentExecutor(
    agent=agent,
    tools=tools,
    verbose=True,  # 打印执行过程
    max_iterations=5
)

# 运行
result = agent_executor.invoke({
    "input": "纽约今天的天气怎么样？另外帮我查一下苹果公司今天的股价。"
})

print(result["output"])
```

## Chain 用法

### LLMChain（最常用的 Chain）

```python
from langchain_openai import ChatOpenAI
from langchain.prompts import PromptTemplate
from langchain.chains import LLMChain

llm = ChatOpenAI(model="gpt-4o")

# 创建 Chain
chain = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template("用一句话解释{concept}")
)

# 运行
result = chain.run(concept="量子计算")
print(result)
```

### SequentialChain（顺序执行）

```python
from langchain.chains import LLMChain, SequentialChain

# Chain 1：提取关键词
chain1 = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template(
        "从以下文本中提取3个关键词：{text}",
        output_variables=["keywords"]
    ),
    output_variables=["keywords"]
)

# Chain 2：根据关键词生成问题
chain2 = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template(
        "根据关键词{keywords}生成3个相关问题",
        output_variables=["questions"]
    ),
    output_variables=["questions"]
)

# 组合成 SequentialChain
overall_chain = SequentialChain(
    chains=[chain1, chain2],
    input_variables=["text"],
    output_variables=["keywords", "questions"],
    verbose=True
)

result = overall_chain({"text": "机器学习是人工智能的一个分支"})
# result: {"text": "...", "keywords": "机器学习, 人工智能, 深度学习", "questions": "..."}
```

### RouterChain（路由分发）

```python
from langchain.chains import LLMChain
from langchain.chains.router import MultiPromptChain
from langchain.prompts import PromptTemplate

# 不同场景的 Prompt
physics_prompt = PromptTemplate.from_template(
    "回答物理问题：{input}",
    output_variables=["text"]
)

math_prompt = PromptTemplate.from_template(
    "回答数学问题：{input}",
    output_variables=["text"]
)

# 路由 Chain
router_chain = MultiPromptChain(
    router_chain=LLMChain(
        llm=llm,
        prompt=PromptTemplate.from_template(
            """根据问题类型选择对应的Chain。

            问题：{input}
            类型：物理/数学/通用

            只返回 Chain 名称：physics/math/general
            """
        )
    ),
    destination_chains={
        "physics": LLMChain(llm=llm, prompt=physics_prompt),
        "math": LLMChain(llm=llm, prompt=math_prompt),
    },
    default_chain=LLMChain(llm=llm, prompt=PromptTemplate.from_template("{input}"))
)
```

## Memory（记忆）

### 对话记忆

```python
from langchain.memory import ConversationBufferMemory
from langchain.chains import LLMChain
from langchain.prompts import PromptTemplate

# 创建带记忆的 Chain
memory = ConversationBufferMemory(memory_key="chat_history", return_messages=True)

chain = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template("{chat_history}\nHuman: {input}\nAI:"),
    memory=memory
)

# 对话
result1 = chain.run("我叫张三")
print(result1)  # 好的，张三，很高兴认识你

result2 = chain.run("我叫什么名字？")
print(result2)  # 你叫张三
```

### 总结记忆

```python
from langchain.memory import ConversationSummaryMemory

memory = ConversationSummaryMemory(llm=llm)

chain = LLMChain(
    llm=llm,
    prompt=PromptTemplate.from_template("{chat_history}\nHuman: {input}\nAI:"),
    memory=memory
)

# 长对话后，记忆会被自动总结，不会无限增长
for i in range(10):
    chain.run(f"这是第{i+1}轮对话")

# memory.buffer 包含的是总结后的内容，而不是完整对话
print(memory.buffer)
```

## Callback 与可观测性

```python
from langchain.callbacks import CallbackManager, StdOutCallbackHandler
from langchain.schema import AgentAction, AgentFinish

# 方式1：全局 Callback
llm = ChatOpenAI(
    model="gpt-4o",
    callback_manager=CallbackManager([StdOutCallbackHandler()])
)

# 方式2：Chain 级 Callback
from langchain.callbacks import get_openai_callback

with get_openai_callback() as cb:
    chain = prompt | llm | StrOutputParser()
    result = chain.invoke({"text": "你好"})

    print(f"总 Token: {cb.total_tokens}")
    print(f"消耗: ${cb.total_cost}")
```

## 本节小结

**LangChain 核心**：
- **LCEL**：用 `|` 符号声明式组合组件
- **Model I/O**：PromptTemplate + LLM + OutputParser
- **Chains**：LLMChain（单个 LLM）、SequentialChain（顺序执行）
- **Tools**：`@tool` 装饰器或 `StructuredTool`
- **Agents**：`create_openai_functions_agent` + `AgentExecutor`
- **Memory**：ConversationBufferMemory、ConversationSummaryMemory

**高级用法**：
- RouterChain：多场景路由
- Callback：执行过程监控
- Streaming：实时输出
