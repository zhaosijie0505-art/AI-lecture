# 第六节 实战：构建企业知识库问答

## 项目需求

> 构建一个企业知识库问答系统：
> - 支持上传多种格式文档（PDF、Word、Markdown）
> - 支持语义搜索和关键词搜索
> - 支持追问和多轮对话
> - 回答带引用来源
> - 支持中文

#### rag/indexer.py

```python
from llama_index import SimpleDirectoryReader
from llama_index.node_parser import HierarchicalNodeParser
from llama_index import VectorStoreIndex, ServiceContext
from llama_index.storage import StorageContext
from llama_index.llms import OpenAI
from llama_index.embeddings import OpenAIEmbedding
import chromadb
from config import config

class DocumentIndexer:
    """文档索引器"""

    def __init__(self):
        self.llm = OpenAI(model=config.llm_model, api_key=config.openai_api_key)
        self.embedding = OpenAIEmbedding(
            model=config.embedding_model,
            api_key=config.openai_api_key
        )
        self.service_context = ServiceContext(
            llm=self.llm,
            embed_model=self.embedding,
            chunk_size=config.chunk_size
        )
        self.index = None
        self._init_vector_store()

    def _init_vector_store(self):
        """初始化向量数据库"""
        self.chroma_client = chromadb.PersistentClient(
            path=config.persist_directory
        )
        self.collection = self.chroma_client.get_or_create_collection(
            name="knowledge_base"
        )

    def ingest(self, file_path: str):
        """索引文档"""
        # 加载文档
        loader = SimpleDirectoryReader(input_files=[file_path])
        documents = loader.load_data()

        # 使用 HierarchicalNodeParser（保留层级结构）
        node_parser = HierarchicalNodeParser.from_defaults(
            chunk_sizes=[2048, 512, 128]  # 大块→中块→小块
        )
        nodes = node_parser.get_nodes_from_documents(documents)

        # 构建索引
        storage_context = StorageContext.from_defaults(
            vector_store=self.collection
        )

        self.index = VectorStoreIndex(
            nodes,
            service_context=self.service_context,
            storage_context=storage_context
        )

        print(f"索引完成：{len(nodes)} 个节点")

    def ingest_directory(self, directory: str):
        """索引目录下所有文档"""
        loader = SimpleDirectoryReader(
            input_dir=directory,
            required_exts=[".pdf", ".docx", ".md", ".txt"]
        )
        documents = loader.load_data()

        node_parser = HierarchicalNodeParser.from_defaults(
            chunk_sizes=[2048, 512, 128]
        )
        nodes = node_parser.get_nodes_from_documents(documents)

        storage_context = StorageContext.from_defaults(
            vector_store=self.collection
        )

        self.index = VectorStoreIndex(
            nodes,
            service_context=self.service_context,
            storage_context=storage_context
        )

        return len(nodes)
```

#### rag/generator.py

```python
from llama_index import VectorStoreIndex
from llama_index.query_engine import RetrieverQueryEngine
from llama_index.retrievers import RecursiveRetriever
from llama_index.postprocessor import SimilarityPostprocessor, RerankPostprocessor
from langchain_openai import ChatOpenAI
from langchain.prompts import ChatPromptTemplate
from config import config

class AnswerGenerator:
    """答案生成器"""

    def __init__(self, index: VectorStoreIndex):
        self.index = index
        self.llm = ChatOpenAI(
            model=config.llm_model,
            api_key=config.openai_api_key,
            temperature=0.3
        )
        self._setup_query_engine()

    def _setup_query_engine(self):
        """设置查询引擎"""
        # 检索器
        retriever = self.index.as_retriever(
            similarity_top_k=config.similarity_top_k
        )

        # 后处理器（Rerank）
        postprocessors = [
            SimilarityPostprocessor(threshold=0.7),
            RerankPostprocessor(
                top_n=config.rerank_top_k,
                model="cross-encoder/ms-marco-MiniLM-L-6-v2"
            )
        ]

        self.query_engine = RetrieverQueryEngine.from_args(
            retriever=retriever,
            node_postprocessors=postprocessors,
            service_context=self.index.service_context
        )

    def generate(self, query: str, chat_history: list = None) -> dict:
        """
        生成回答

        Returns:
            {"answer": str, "sources": [{"content": str, "score": float}]}
        """
        # 检索相关文档
        nodes = self.query_engine.retrieve(query)

        # 构建上下文（带来源标注）
        context_parts = []
        sources = []
        for i, node in enumerate(nodes, 1):
            context_parts.append(f"[来源{i}] {node.text}")
            sources.append({
                "content": node.text[:200] + "..." if len(node.text) > 200 else node.text,
                "score": node.score if hasattr(node, 'score') else None
            })

        context = "\n\n".join(context_parts)

        # 构建 Prompt
        prompt = self._build_prompt(query, context, chat_history)

        # 生成回答
        answer = self.llm.invoke(prompt).content

        return {
            "answer": answer,
            "sources": sources
        }

    def _build_prompt(self, query: str, context: str, history: list) -> str:
        """构建 Prompt"""
        history_text = ""
        if history:
            history_text = "\n\n对话历史：\n" + "\n".join([
                f"用户：{h['user']}\n助手：{h['assistant']}"
                for h in history[-3:]  # 最近3轮
            ])

        return f"""你是一个企业知识库问答助手。请根据以下参考资料回答用户问题。

参考资料：
{context}
{history_text}

用户问题：{query}

回答要求：
1. 基于给定的参考资料回答
2. 在回答中用 [来源N] 标注信息来源
3. 如果参考资料中没有相关信息，直接说你不知道
4. 回答要简洁、准确

回答："""
```

#### rag/chat.py

```python
from typing import List, Dict
from dataclasses import dataclass, field
from config import config

@dataclass
class ChatMessage:
    role: str  # "user" or "assistant"
    content: str

@dataclass
class ChatSession:
    session_id: str
    history: List[ChatMessage] = field(default_factory=list)
    created_at: str = ""

class ChatManager:
    """对话管理器"""

    def __init__(self):
        self.sessions: Dict[str, ChatSession] = {}

    def get_or_create_session(self, session_id: str) -> ChatSession:
        if session_id not in self.sessions:
            self.sessions[session_id] = ChatSession(session_id=session_id)
        return self.sessions[session_id]

    def add_message(self, session_id: str, role: str, content: str):
        session = self.get_or_create_session(session_id)
        session.history.append(ChatMessage(role=role, content=content))

        # 限制历史长度
        if len(session.history) > config.max_history_turns * 2:
            session.history = session.history[-config.max_history_turns * 2:]

    def get_history(self, session_id: str) -> List[Dict]:
        session = self.get_or_create_session(session_id)
        return [
            {"role": msg.role, "content": msg.content}
            for msg in session.history
        ]
```

### 3. API 层（FastAPI）

```python
from fastapi import FastAPI, UploadFile, File, HTTPException
from fastapi.middleware.cors import CORSMiddleware
import tempfile
import os
from pydantic import BaseModel

from rag.indexer import DocumentIndexer
from rag.generator import AnswerGenerator
from rag.chat import ChatManager

app = FastAPI(title="企业知识库问答 API")

# CORS
app.add_middleware(
    CORSMiddleware,
    allow_origins=["*"],
    allow_credentials=True,
    allow_methods=["*"],
    allow_headers=["*"],
)

# 全局组件
indexer = DocumentIndexer()
chat_manager = ChatManager()

# 请求/响应模型
class QueryRequest(BaseModel):
    question: str
    session_id: str = "default"

class QueryResponse(BaseModel):
    answer: str
    sources: list
    session_id: str

@app.post("/upload")
async def upload_document(file: UploadFile = File(...)):
    """上传并索引文档"""
    # 保存临时文件
    with tempfile.NamedTemporaryFile(
        delete=False,
        suffix=file.filename
    ) as tmp:
        content = await file.read()
        tmp.write(content)
        tmp_path = tmp.name

    try:
        # 索引文档
        indexer.ingest(tmp_path)
        return {"status": "success", "message": f"已索引文档：{file.filename}"}
    except Exception as e:
        raise HTTPException(status_code=500, detail=str(e))
    finally:
        os.unlink(tmp_path)

@app.post("/query", response_model=QueryResponse)
async def query(request: QueryRequest):
    """问答"""
    if indexer.index is None:
        raise HTTPException(
            status_code=400,
            detail="请先上传文档"
        )

    # 获取对话历史
    history = chat_manager.get_history(request.session_id)

    # 生成回答
    generator = AnswerGenerator(indexer.index)
    result = generator.generate(request.question, history)

    # 保存对话历史
    chat_manager.add_message(request.session_id, "user", request.question)
    chat_manager.add_message(request.session_id, "assistant", result["answer"])

    return QueryResponse(
        answer=result["answer"],
        sources=result["sources"],
        session_id=request.session_id
    )

if __name__ == "__main__":
    import uvicorn
    uvicorn.run(app, host="0.0.0.0", port=8000)
```

### 4. 前端（Streamlit）

```python
import streamlit as st
import requests
import uuid

st.set_page_config(page_title="企业知识库问答", page_icon="📚")

st.title("📚 企业知识库问答")

# 侧边栏：文档上传
with st.sidebar:
    st.header("文档管理")
    uploaded_file = st.file_uploader(
        "上传文档",
        type=["pdf", "docx", "md", "txt"]
    )
    if uploaded_file:
        if st.button("索引文档"):
            with st.spinner("正在索引..."):
                files = {"file": (uploaded_file.name, uploaded_file.getvalue())}
                response = requests.post(
                    "http://localhost:8000/upload",
                    files=files
                )
                if response.status_code == 200:
                    st.success("文档索引成功！")
                else:
                    st.error(f"索引失败：{response.text}")

# 主界面：问答
if "session_id" not in st.session_state:
    st.session_state.session_id = str(uuid.uuid4())

question = st.text_input("请输入您的问题：", placeholder="例如：公司的年假政策是什么？")

if question:
    if st.button("提问"):
        with st.spinner("思考中..."):
            response = requests.post(
                "http://localhost:8000/query",
                json={"question": question, "session_id": st.session_state.session_id}
            )

            if response.status_code == 200:
                result = response.json()

                st.markdown("**回答：**")
                st.write(result["answer"])

                with st.expander("📖 参考来源"):
                    for i, source in enumerate(result["sources"], 1):
                        st.markdown(f"**来源 {i}** (相似度: {source['score']:.2f})")
                        st.write(source["content"])
                        st.divider()
            else:
                st.error(f"请求失败：{response.text}")

# 对话历史
st.divider()
st.markdown("**对话历史**")
```

## 本节小结

**项目结构**：
- `indexer.py`：文档加载、解析、索引
- `generator.py`：检索、Rerank、生成回答
- `chat.py`：对话状态管理

**技术选型**：
- 索引：LlamaIndex + HierarchicalNodeParser
- 检索：VectorStoreIndex + RerankPostprocessor
- 生成：LangChain + ChatOpenAI
- 持久化：ChromaDB

**API**：FastAPI 提供 `/upload` 和 `/query` 接口

**前端**：Streamlit 快速构建 demo 界面
