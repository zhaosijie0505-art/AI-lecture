# 第二节 多跳推理与链式检索

## 什么是多跳推理

简单问题（0-hop）：直接检索答案
```
"特斯拉的 CEO 是谁？" → 检索"特斯拉 CEO" → 回答
```

单跳问题（1-hop）：检索一次获得答案
```
"特斯拉的市值是多少？" → 检索"特斯拉市值" → 回答
```

多跳问题（2+ hops）：需要多个推理步骤
```
"特斯拉 CEO 的出生地是哪里？"

Step 1: 检索"特斯拉 CEO" → 马斯克
Step 2: 检索"马斯克 出生地" → 洛杉矶
回答：洛杉矶
```

```
多跳问题的难点：

1. 不知道需要多少跳
2. 每一步的检索 query 可能不清楚
3. 错误会累积（一步错，步步错）
```

## 1. Chain of Thought Retrieval（思维链检索）

### 核心思想

让模型显式地推理出中间问题，再逐步检索：

```python
cot_prompt = PromptTemplate.from_template("""
问题：{question}

请逐步推理这个问题需要哪些信息：

Step 1: 识别问题中需要查找的关键实体或概念
→ 实体1：...

Step 2: 如果需要，基于 Step 1 的结果推断下一步
→ 需要查找：...

Step 3: 持续推理直到可以回答问题

最终回答需要的检索词：[列出检索词]

请一步一步思考：""")

cot_chain = LLMChain(llm=llm, prompt=cot_prompt)

def cot_retrieve(question: str, vector_store, k: int = 5) -> dict:
    """
    思维链检索

    1. 让 LLM 显式推理出需要检索的内容
    2. 按顺序检索
    3. 收集所有中间结果
    """
    # Step 1: 推理需要检索什么
    reasoning = cot_chain.run(question)

    # 解析推理结果，提取需要检索的 query
    # 这里简化处理，实际可以更复杂地解析
    retrieval_queries = extract_retrieval_queries(reasoning)

    print(f"推理出 {len(retrieval_queries)} 个检索步骤：")
    for i, q in enumerate(retrieval_queries, 1):
        print(f"  步骤{i}：{q}")

    # Step 2: 按顺序检索
    intermediate_results = []
    for query in retrieval_queries:
        results = vector_store.similarity_search(query, k=k)
        intermediate_results.append({
            "query": query,
            "results": results
        })

    return {
        "reasoning": reasoning,
        "intermediate_results": intermediate_results
    }

def extract_retrieval_queries(reasoning: str) -> list:
    """从推理文本中提取检索 query"""
    # 简化版：找"检索"、"查找"等关键词后面的内容
    import re
    queries = re.findall(r'检索[：:]?\s*(.+?)(?:\n|$)', reasoning)
    if not queries:
        queries = re.findall(r'查找[：:]?\s*(.+?)(?:\n|$)', reasoning)
    if not queries:
        queries = [q.strip() for q in reasoning.split('\n') if q.strip()]
    return queries[:5]  # 最多 5 个
```

### 完整多跳问答

```python
def multi_hop_answer(question: str, vector_store, llm, k: int = 5) -> str:
    """
    多跳问答完整流程
    """
    # Step 1: 思维链推理
    cot_result = cot_retrieve(question, vector_store, k=k)

    # Step 2: 收集所有中间信息
    context_parts = []
    context_parts.append(f"问题：{question}\n")

    for i, step_result in enumerate(cot_result["intermediate_results"], 1):
        context_parts.append(f"\n--- 步骤{i}：{step_result['query']} ---")
        for j, (doc, score) in enumerate(step_result["results"][:2], 1):
            context_parts.append(f"[结果{j}] {doc.page_content}")

    context = "\n".join(context_parts)

    # Step 3: 生成回答
    answer_prompt = f"""
基于以下中间检索结果，逐步推理回答问题。

{context}

请逐步推理，然后给出最终答案：
"""

    answer = llm.invoke(answer_prompt)
    return answer.content
```

## 2. Forward/Backward Chain Retrieval

### 核心思想

类似知识图谱的推理方式：
- **Forward Chain**：从已知事实出发，推理新事实
- **Backward Chain**：从目标问题出发，反向找所需事实

```python
# 知识图谱风格的链式推理
# 假设我们有 (entity, relation, entity) 三元组

knowledge_graph = {
    ("特斯拉", "CEO", "马斯克"),
    ("特斯拉", "总部", "德克萨斯"),
    ("马斯克", "出生地", "比勒陀利亚"),
    ("马斯克", "出生地", "洛杉矶"),  # 迁移后
    ("马斯克", "公司", "SpaceX"),
}

def forward_chain(question: str, knowledge_graph: set) -> list:
    """
    前向链：从事实出发推理

    问题："特斯拉的 CEO 出生在哪里？"

    推理路径：
    特斯拉 → CEO → 马斯克
    马斯克 → 出生地 → ?

    找到答案：洛杉矶
    """
    # 简化实现
    reasoning_steps = []

    # 提取问题中的实体和关系
    entities = extract_entities(question)

    for entity in entities:
        # 前向扩展：从已知实体出发
        for s, p, o in knowledge_graph:
            if s == entity or o == entity:
                reasoning_steps.append((s, p, o))

    return reasoning_steps

def backward_chain(question: str, knowledge_graph: set) -> list:
    """
    后向链：从目标反推

    问题需要回答的 target：出生地
    从 target 反推需要哪些事实
    """
    target = extract_target(question)  # 例如：出生地

    # 找与 target 相关的关系
    relevant_facts = []
    for s, p, o in knowledge_graph:
        if p == target:
            relevant_facts.append((s, p, o))

    return relevant_facts
```

### 结合 RAG 的链式检索

```python
class ChainRetrieval:
    """结合 RAG 的链式检索"""

    def __init__(self, vector_store, llm):
        self.vector_store = vector_store
        self.llm = llm
        self.max_hops = 5

    def retrieve_chain(self, question: str) -> list:
        """
        多跳检索链

        返回：[(query, results), ...]
        """
        chain = []
        current_question = question

        for hop in range(self.max_hops):
            print(f"\n=== 第 {hop+1} 跳 ===")
            print(f"当前问题：{current_question}")

            # 检索
            results = self.vector_store.similarity_search(current_question, k=5)

            # 检查是否找到答案
            if self._contains_answer(results, question):
                print("找到答案，停止检索")
                chain.append((current_question, results))
                break

            chain.append((current_question, results))

            # 生成下一跳问题
            current_question = self._generate_next_question(
                question, chain
            )

            if not current_question:
                print("无法继续推理，停止")
                break

        return chain

    def _contains_answer(self, results, question) -> bool:
        """检查结果是否包含答案（简化判断）"""
        # 实际可以用更复杂的方法，比如让 LLM 判断
        return False  # 保守估计

    def _generate_next_question(self, original_question: str, chain: list) -> str:
        """生成下一跳问题"""
        prompt = f"""
原始问题：{original_question}

已完成检索：
"""

        for i, (q, results) in enumerate(chain):
            prompt += f"\n步骤{i+1}：{q}"
            for j, (doc, _) in enumerate(results[:2], 1):
                prompt += f"\n  结果{j}：{doc.page_content[:100]}..."

        prompt += """
\n基于以上检索结果，还需要查找什么信息才能回答原始问题？
如果问题已可回答，返回"已完成"。
否则，提出下一个需要检索的问题（30字以内）：
"""

        response = self.llm.invoke(prompt)
        next_q = response.content.strip()

        if "已完成" in next_q:
            return None

        return next_q
```

## 3. Iteration RAG（迭代检索）

### 核心思想

不只检索一次，而是多轮迭代：
- 每轮检索后，评估当前结果是否足够回答问题
- 如果不够，生成新的检索 query，继续检索

```python
class IterativeRAG:
    """迭代式 RAG"""

    def __init__(self, vector_store, llm, max_iterations: int = 3):
        self.vector_store = vector_store
        self.llm = llm
        self.max_iterations = max_iterations

    def query(self, question: str) -> dict:
        """
        迭代检索直到有足够信息
        """
        all_context = []
        current_iteration = 0

        while current_iteration < self.max_iterations:
            print(f"\n=== 迭代 {current_iteration + 1} ===")

            # 如果是第一轮，直接用原问题
            if current_iteration == 0:
                query = question
            else:
                # 生成新的检索 query
                query = self._refine_query(question, all_context)
                if not query:
                    print("无法生成新的检索 query，停止")
                    break

            # 检索
            results = self.vector_store.similarity_search(query, k=5)

            # 添加到上下文
            for doc, score in results:
                all_context.append({
                    "content": doc.page_content,
                    "score": score,
                    "iteration": current_iteration
                })

            # 检查是否足够
            if self._is_sufficient(question, all_context):
                print("信息足够，停止检索")
                break

            current_iteration += 1

        # 生成最终答案
        answer = self._generate_answer(question, all_context)

        return {
            "answer": answer,
            "context": all_context,
            "iterations": current_iteration + 1
        }

    def _refine_query(self, question: str, context: list) -> str:
        """基于已有上下文，生成更精准的检索 query"""
        context_summary = "\n".join([
            f"- {item['content'][:100]}..."
            for item in context[-5:]  # 最近 5 条
        ])

        prompt = f"""
原始问题：{question}

已有信息：
{context_summary}

请基于已有信息，生成一个更精准的检索 query 来补充缺失的信息。
只返回一个检索 query，30字以内。
"""

        response = self.llm.invoke(prompt)
        return response.content.strip()

    def _is_sufficient(self, question: str, context: list) -> bool:
        """判断已有信息是否足够回答问题"""
        context_text = "\n".join([item["content"] for item in context])

        prompt = f"""
问题：{question}

已有信息：
{context_text}

判断：这些信息是否足够完整回答问题？
回答：是/否，并简要说明原因。
"""

        response = self.llm.invoke(prompt)

        # 如果是肯定回答（简化判断）
        if "是" in response.content[:5]:
            return True
        return False

    def _generate_answer(self, question: str, context: list) -> str:
        """生成最终答案"""
        context_text = "\n".join([
            f"[来源{i+1}] {item['content']}"
            for i, item in enumerate(context[:5])
        ])

        prompt = f"""
基于以下信息回答问题。

问题：{question}

信息：
{context_text}

回答：
"""

        return self.llm.invoke(prompt).content
```

## 4. 知识图谱增强检索

### 核心思想

在 RAG 中引入知识图谱，利用图结构做关系推理：

```python
# 简化版：实体关系三元组检索

class GraphAugmentedRAG:
    """知识图谱增强的 RAG"""

    def __init__(self, vector_store, knowledge_graph: dict = None):
        self.vector_store = vector_store
        self.knowledge_graph = knowledge_graph or {}

    def query(self, question: str) -> dict:
        """
        混合检索：向量搜索 + 知识图谱推理
        """
        # 1. 实体识别
        entities = self._extract_entities(question)

        # 2. 向量检索
        vector_results = self.vector_store.similarity_search(question, k=5)

        # 3. 知识图谱检索
        kg_results = []
        for entity in entities:
            # 找与实体相关的关系
            related = self._get_related_from_kg(entity)
            kg_results.extend(related)

        # 4. 合并结果
        merged = self._merge_results(vector_results, kg_results)

        return {
            "vector_results": vector_results,
            "kg_results": kg_results,
            "merged_results": merged
        }

    def _extract_entities(self, text: str) -> list:
        """从文本中提取实体（简化版）"""
        # 实际可以用 NER 模型
        return [word for word in text if len(word) > 2]

    def _get_related_from_kg(self, entity: str) -> list:
        """从知识图谱获取相关信息"""
        results = []
        for s, p, o in self.knowledge_graph:
            if s == entity or o == entity:
                results.append({
                    "subject": s,
                    "predicate": p,
                    "object": o
                })
        return results

    def _merge_results(self, vector_results: list, kg_results: list) -> list:
        """合并向量检索和知识图谱检索结果"""
        # 简单策略：都保留，标注来源
        merged = []

        seen = set()
        for doc, score in vector_results:
            if doc.page_content not in seen:
                merged.append({"doc": doc, "score": score, "source": "vector"})
                seen.add(doc.page_content)

        for kg_item in kg_results:
            text = f"{kg_item['subject']} {kg_item['predicate']} {kg_item['object']}"
            if text not in seen:
                merged.append({"doc": type('Doc', (), {'page_content': text})(), "score": 1.0, "source": "kg"})
                seen.add(text)

        return merged
```

## 本节小结

**多跳推理的四种方法**：

| 方法 | 思想 | 适用场景 |
|------|------|---------|
| **CoT Retrieval** | 显式推理中间问题，逐步检索 | 需要多步推理的复杂问题 |
| **Forward/Backward Chain** | 从事实出发/从目标反推 | 有结构化知识的场景 |
| **Iterative RAG** | 多轮迭代直到信息足够 | 信息分散、难以一次找到 |
| **Graph RAG** | 引入知识图谱做关系推理 | 需要实体关系的复杂推理 |

**实战要点**：
- 多跳推理要限制跳数（max_hops=3-5），防止无限循环
- 每跳后评估是否已找到答案，避免不必要的后续检索
- 迭代检索要有终止条件，防止资源浪费
