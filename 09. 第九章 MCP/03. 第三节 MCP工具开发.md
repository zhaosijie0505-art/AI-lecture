## 工具开发

### 1. 基础工具

```python
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent
import asyncio

app = Server("my-server")

# 定义工具列表
@app.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="hello",
            description="发送问候语",
            inputSchema={
                "type": "object",
                "properties": {
                    "name": {
                        "type": "string",
                        "description": "被问候的人名"
                    },
                    "language": {
                        "type": "string",
                        "description": "语言（zh/en/ja）",
                        "default": "zh"
                    }
                },
                "required": ["name"]
            }
        )
    ]

# 处理工具调用
@app.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "hello":
        name_arg = arguments["name"]
        lang = arguments.get("language", "zh")

        greetings = {
            "zh": f"你好，{name_arg}！",
            "en": f"Hello, {name_arg}!",
            "ja": f"こんにちは、{name_arg}さん！"
        }

        message = greetings.get(lang, greetings["zh"])

        return [TextContent(type="text", text=message)]

    raise ValueError(f"Unknown tool: {name}")

async def main():
    async with stdio_server() as (read_stream, write_stream):
        await app.run(
            read_stream,
            write_stream,
            app.create_initialization_options()
        )

if __name__ == "__main__":
    asyncio.run(main())
```

### 2. 复杂工具（带文件操作）

```python
import os
import json
from pathlib import Path

@app.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="read_file",
            description="读取文件内容",
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "文件路径"
                    },
                    "max_lines": {
                        "type": "number",
                        "description": "最大读取行数",
                        "default": 100
                    }
                },
                "required": ["path"]
            }
        ),
        Tool(
            name="write_file",
            description="写入文件内容",
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "文件路径"
                    },
                    "content": {
                        "type": "string",
                        "description": "文件内容"
                    },
                    "append": {
                        "type": "boolean",
                        "description": "是否追加模式",
                        "default": False
                    }
                },
                "required": ["path", "content"]
            }
        ),
        Tool(
            name="list_directory",
            description="列出目录内容",
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "目录路径"
                    },
                    "include_hidden": {
                        "type": "boolean",
                        "description": "是否包含隐藏文件",
                        "default": False
                    }
                },
                "required": ["path"]
            }
        )
    ]

@app.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if name == "read_file":
        path = Path(arguments["path"])
        max_lines = arguments.get("max_lines", 100)

        if not path.exists():
            return [TextContent(type="text", text=f"文件不存在: {path}")]

        content = path.read_text(encoding="utf-8")
        lines = content.split("\n")

        if len(lines) > max_lines:
            content = "\n".join(lines[:max_lines])
            content += f"\n\n... (共 {len(lines)} 行，已截断)"

        return [TextContent(type="text", text=content)]

    elif name == "write_file":
        path = Path(arguments["path"])
        content = arguments["content"]
        append = arguments.get("append", False)

        path.parent.mkdir(parents=True, exist_ok=True)

        mode = "a" if append else "w"
        path.write_text(content, encoding="utf-8", errors="ignore")

        action = "追加写入" if append else "写入"
        return [TextContent(type="text", text=f"{action}成功: {path}")]

    elif name == "list_directory":
        path = Path(arguments["path"])
        include_hidden = arguments.get("include_hidden", False)

        if not path.exists():
            return [TextContent(type="text", text=f"目录不存在: {path}")]

        if not path.is_dir():
            return [TextContent(type="text", text=f"不是目录: {path}")]

        items = []
        for item in path.iterdir():
            if not include_hidden and item.name.startswith("."):
                continue

            item_type = "📁" if item.is_dir() else "📄"
            size = item.stat().st_size if item.is_file() else 0
            items.append(f"{item_type} {item.name} ({size} bytes)")

        if not items:
            return [TextContent(type="text", text="目录为空")]

        return [TextContent(type="text", text="\n".join(items))]

    raise ValueError(f"Unknown tool: {name}")
```

### 3. 外部 API 工具

```python
import httpx
from typing import Optional

class APIClient:
    def __init__(self, api_key: str):
        self.api_key = api_key
        self.base_url = "https://api.example.com/v1"
        self.client = httpx.AsyncClient(timeout=30)

    async def search(self, query: str, limit: int = 10) -> dict:
        """搜索 API"""
        response = await self.client.get(
            f"{self.base_url}/search",
            params={"q": query, "limit": limit},
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        response.raise_for_status()
        return response.json()

    async def get_weather(self, city: str) -> dict:
        """获取天气"""
        response = await self.client.get(
            f"{self.base_url}/weather",
            params={"city": city},
            headers={"Authorization": f"Bearer {self.api_key}"}
        )
        response.raise_for_status()
        return response.json()

# 全局 API Client（在 main 中初始化）
api_client: Optional[APIClient] = None

def init_api_client(api_key: str):
    global api_client
    api_client = APIClient(api_key)

@app.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="search",
            description="搜索信息",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "搜索关键词"
                    },
                    "limit": {
                        "type": "number",
                        "description": "返回结果数量",
                        "default": 10
                    }
                },
                "required": ["query"]
            }
        ),
        Tool(
            name="weather",
            description="获取天气信息",
            inputSchema={
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称"
                    }
                },
                "required": ["city"]
            }
        )
    ]

@app.call_tool()
async def call_tool(name: str, arguments: dict) -> list[TextContent]:
    if not api_client:
        return [TextContent(type="text", text="API Client 未初始化")]

    if name == "search":
        results = await api_client.search(
            query=arguments["query"],
            limit=arguments.get("limit", 10)
        )
        return [TextContent(type="text", text=json.dumps(results, ensure_ascii=False))]

    elif name == "weather":
        weather = await api_client.get_weather(city=arguments["city"])
        return [TextContent(type="text", text=json.dumps(weather, ensure_ascii=False))]

    raise ValueError(f"Unknown tool: {name}")
```

            name="status",

@app.read_resource()
async def read_resource(uri: AnyUrl) -> str:
    """读取资源"""
    uri_str = str(uri)

    if uri_str.startswith("file://"):
        path = uri_str[7:]  # 去掉 file://
        return Path(path).read_text(encoding="utf-8")

    elif uri_str.startswith("memory://"):
        key = uri_str[9:]  # 去掉 memory://
        return get_memory_item(key)  # 自定义实现

    raise ValueError(f"Unknown resource URI: {uri}")

## 完整项目示例：天气 MCP Server

```python
#!/usr/bin/env python3
"""
天气 MCP Server

功能：
- 获取当前天气
- 获取天气预报
- 获取空气质量
"""

import httpx
import asyncio
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent
from typing import Any
import json

# 服务器实例
app = Server("weather-mcp-server")

# API 配置（通过环境变量或配置文件设置）
WEATHER_API_KEY = os.getenv("WEATHER_API_KEY", "")
WEATHER_API_BASE = "https://api.weatherapi.com/v1"

# API Client
async def get_weather(city: str) -> dict:
    """获取当前天气"""
    async with httpx.AsyncClient() as client:
        response = await client.get(
            f"{WEATHER_API_BASE}/current.json",
            params={
                "key": WEATHER_API_KEY,
                "q": city,
                "aqi": "no"
            }
        )
        response.raise_for_status()
        return response.json()

async def get_forecast(city: str, days: int = 3) -> dict:
    """获取天气预报"""
    async with httpx.AsyncClient() as client:
        response = await client.get(
            f"{WEATHER_API_BASE}/forecast.json",
            params={
                "key": WEATHER_API_KEY,
                "q": city,
                "days": days,
                "aqi": "no",
                "alerts": "no"
            }
        )
        response.raise_for_status()
        return response.json()

# 工具定义
@app.list_tools()
async def list_tools() -> list[Tool]:
    return [
        Tool(
            name="current_weather",
            description="获取城市当前天气",
            inputSchema={
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称或坐标（如：北京 或 39.9,116.4）"
                    },
                    "aqi": {
                        "type": "boolean",
                        "description": "是否包含空气质量",
                        "default": False
                    }
                },
                "required": ["city"]
            }
        ),
        Tool(
            name="weather_forecast",
            description="获取天气预报",
            inputSchema={
                "type": "object",
                "properties": {
                    "city": {
                        "type": "string",
                        "description": "城市名称"
                    },
                    "days": {
                        "type": "number",
                        "description": "预报天数（1-10）",
                        "default": 3
                    }
                },
                "required": ["city"]
            }
        )
    ]

# 工具调用处理
@app.call_tool()
async def call_tool(name: str, arguments: Any) -> list[TextContent]:
    try:
        if name == "current_weather":
            city = arguments["city"]
            include_aqi = arguments.get("aqi", False)

            weather = await get_weather(city)

            result = weather["current"]
            location = weather["location"]

            output = f"""📍 {location['name']}, {location['country']}
🌡️ 温度：{result['temp_c']}°C（体感 {result['feelslike_c']}°C）
🌤️ 天气：{result['condition']['text']}
💨 风速：{result['wind_kph']} km/h
💧 湿度：{result['humidity']}%
"""

            if include_aqi:
                output += f"😷 空气质量：{result.get('air_quality', {}).get('pm2.5', 'N/A')}"

            return [TextContent(type="text", text=output)]

        elif name == "weather_forecast":
            city = arguments["city"]
            days = arguments.get("days", 3)

            forecast = await get_forecast(city, days)

            location = forecast["location"]
            output = f"📍 {location['name']}, {location['country']}\n\n"

            for day in forecast["forecast"]["forecastday"]:
                date = day["date"]
                avg_temp = day["day"]["avgtemp_c"]
                condition = day["day"]["condition"]["text"]
                rain_chance = day["day"]["daily_chance_of_rain"]

                output += f"📅 {date}\n"
                output += f"   🌡️ 平均温度：{avg_temp}°C\n"
                output += f"   🌤️ 天气：{condition}\n"
                output += f"   💧 降雨概率：{rain_chance}%\n\n"

            return [TextContent(type="text", text=output.strip())]

        else:
            raise ValueError(f"Unknown tool: {name}")

    except Exception as e:
        return [TextContent(type="text", text=f"错误：{str(e)}")]

# 主函数
async def main():
    if not WEATHER_API_KEY:
        print("警告：WEATHER_API_KEY 未设置，部分功能可能不可用", file=sys.stderr)

    async with stdio_server() as (read_stream, write_stream):
        await app.run(
            read_stream,
            write_stream,
            app.create_initialization_options()
        )

if __name__ == "__main__":
    import os
    import sys
    asyncio.run(main())
```

## 本节小结

**工具开发**：
- `@app.list_tools()`：返回可用工具列表
- `@app.call_tool()`：处理工具调用逻辑
- 工具参数使用 JSON Schema 定义

**资源开发**：
- `@app.list_resources()`：返回静态资源
- `@app.list_resource_templates()`：返回动态资源模板
- `@app.read_resource()`：读取资源内容

**Prompt 开发**：
- `@app.list_prompts()`：返回 Prompt 模板列表
- `@app.get_prompt()`：获取具体 Prompt 内容

**最佳实践**：
- 敏感信息（如 API Key）存储在环境变量，不暴露给 Client
- 返回给 Client 的只是业务数据，不包含内部实现细节
- 统一的错误处理，返回友好的错误信息
