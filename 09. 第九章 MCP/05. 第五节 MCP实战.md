#### main.py

```python
#!/usr/bin/env python3
"""
公司内部知识库 MCP Server

功能：
- 搜索公司文档
- 查询员工信息
- 查询项目状态
"""

import asyncio
import os
from pathlib import Path
from mcp.server import Server
from mcp.server.stdio import stdio_server
from mcp.types import Tool, TextContent
from typing import Any

# 初始化 Server
app = Server("company-knowledge-base")

# 数据路径配置
DATA_DIR = Path(__file__).parent / "data"
EMPLOYEES_CSV = DATA_DIR / "employees.csv"
PROJECTS_JSON = DATA_DIR / "projects.json"

# 导入工具
from tools.document_tools import list_document_tools, handle_document_tool
from tools.employee_tools import list_employee_tools, handle_employee_tool
from tools.project_tools import list_project_tools, handle_project_tool

# 注册工具
@app.list_tools()
async def list_tools() -> list[Tool]:
    """返回所有可用工具"""
    return [
        *list_document_tools(),
        *list_employee_tools(),
        *list_project_tools()
    ]

# 处理工具调用
@app.call_tool()
async def call_tool(name: str, arguments: Any) -> list[TextContent]:
    """处理工具调用"""
    # 文档工具
    if name in ["search_documents", "read_document", "list_documents"]:
        return await handle_document_tool(name, arguments)

    # 员工工具
    elif name in ["search_employee", "get_department", "list_departments"]:
        return await handle_employee_tool(name, arguments)

    # 项目工具
    elif name in ["get_project", "list_projects", "get_project_timeline"]:
        return await handle_project_tool(name, arguments)

    raise ValueError(f"Unknown tool: {name}")

# 主函数
async def main():
    # 验证数据文件
    if not DATA_DIR.exists():
        print(f"警告：数据目录不存在: {DATA_DIR}", file=sys.stderr)

    async with stdio_server() as (read_stream, write_stream):
        await app.run(
            read_stream,
            write_stream,
            app.create_initialization_options()
        )

if __name__ == "__main__":
    import sys
    asyncio.run(main())
```

#### tools/document_tools.py

```python
import os
from pathlib import Path
from typing import Any
from mcp.types import Tool, TextContent

# 文档目录配置
DOCS_DIR = Path(os.getenv("COMPANY_DOCS_DIR", "/Users/demo/company-docs"))

def list_document_tools() -> list[Tool]:
    return [
        Tool(
            name="search_documents",
            description="在公司文档库中搜索文档",
            inputSchema={
                "type": "object",
                "properties": {
                    "query": {
                        "type": "string",
                        "description": "搜索关键词"
                    },
                    "file_type": {
                        "type": "string",
                        "description": "文件类型过滤（如：md, pdf, docx）",
                        "default": ""
                    }
                },
                "required": ["query"]
            }
        ),
        Tool(
            name="read_document",
            description="读取指定文档内容",
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "文档路径（相对于文档根目录）"
                    },
                    "max_chars": {
                        "type": "number",
                        "description": "最大读取字符数",
                        "default": 10000
                    }
                },
                "required": ["path"]
            }
        ),
        Tool(
            name="list_documents",
            description="列出文档目录结构",
            inputSchema={
                "type": "object",
                "properties": {
                    "path": {
                        "type": "string",
                        "description": "目录路径（相对于文档根目录）",
                        "default": ""
                    }
                }
            }
        )
    ]

async def handle_document_tool(name: str, arguments: Any) -> list[TextContent]:
    if name == "search_documents":
        query = arguments["query"]
        file_type = arguments.get("file_type", "")

        results = search_docs_recursive(DOCS_DIR, query, file_type)

        if not results:
            return [TextContent(type="text", text="未找到相关文档")]

        output = f"找到 {len(results)} 个相关文档：\n\n"
        for path, preview in results[:10]:
            rel_path = path.relative_to(DOCS_DIR)
            output += f"📄 {rel_path}\n"
            output += f"   预览：{preview[:100]}...\n\n"

        return [TextContent(type="text", text=output)]

    elif name == "read_document":
        doc_path = arguments["path"]
        max_chars = arguments.get("max_chars", 10000)

        full_path = DOCS_DIR / doc_path

        if not full_path.exists():
            return [TextContent(type="text", text=f"文档不存在: {doc_path}")]

        content = full_path.read_text(encoding="utf-8", errors="ignore")

        if len(content) > max_chars:
            content = content[:max_chars] + f"\n\n...(已截断，共 {len(content)} 字符)"

        return [TextContent(type="text", text=content)]

    elif name == "list_documents":
        path = arguments.get("path", "")
        full_path = DOCS_DIR / path if path else DOCS_DIR

        if not full_path.exists():
            return [TextContent(type="text", text=f"目录不存在: {path}")]

        return format_directory_tree(full_path)

    return [TextContent(type="text", text=f"Unknown tool: {name}")]

def search_docs_recursive(directory: Path, query: str, file_type: str = "") -> list:
    """递归搜索文档"""
    results = []
    query_lower = query.lower()

    for item in directory.rglob("*"):
        if item.is_file():
            # 文件类型过滤
            if file_type and item.suffix[1:] != file_type:
                continue

            # 跳过隐藏文件
            if item.name.startswith("."):
                continue

            # 在文件名和内容中搜索
            if query_lower in item.name.lower():
                results.append((item, item.name))
            elif item.suffix in [".md", ".txt", ".pdf"]:
                try:
                    content = item.read_text(encoding="utf-8", errors="ignore")
                    if query_lower in content.lower():
                        # 找到匹配行作为预览
                        for line in content.split("\n"):
                            if query_lower in line.lower():
                                results.append((item, line.strip()))
                                break
                except:
                    pass

    return results

def format_directory_tree(path: Path, prefix: str = "", max_depth: int = 3) -> list[TextContent]:
    """格式化目录树"""
    if not path.is_dir():
        return [TextContent(type="text", text=str(path))]

    items = []
    output_parts = []

    try:
        children = sorted(path.iterdir(), key=lambda x: (not x.is_dir(), x.name))
    except PermissionError:
        return [TextContent(type="text", text="无权限访问")]

    for child in children:
        if child.name.startswith("."):
            continue

        icon = "📁" if child.is_dir() else "📄"
        output_parts.append(f"{prefix}{icon} {child.name}")

        if child.is_dir() and len(prefix) // 4 < max_depth:
            sub_content = format_directory_tree(
                child,
                prefix=prefix + "    ",
                max_depth=max_depth
            )
            if sub_content:
                output_parts.append(sub_content[0].text)

    return [TextContent(type="text", text="\n".join(output_parts))]
```

#### tools/employee_tools.py

```python
import csv
import json
from pathlib import Path
from typing import Any
from mcp.types import Tool, TextContent

EMPLOYEES_CSV = Path(__file__).parent.parent / "data" / "employees.csv"

def list_employee_tools() -> list[Tool]:
    return [
        Tool(
            name="search_employee",
            description="搜索员工信息",
            inputSchema={
                "type": "object",
                "properties": {
                    "name": {
                        "type": "string",
                        "description": "员工姓名"
                    },
                    "department": {
                        "type": "string",
                        "description": "部门名称",
                        "default": ""
                    }
                }
            }
        ),
        Tool(
            name="get_department",
            description="获取部门成员列表",
            inputSchema={
                "type": "object",
                "properties": {
                    "department": {
                        "type": "string",
                        "description": "部门名称"
                    }
                },
                "required": ["department"]
            }
        ),
        Tool(
            name="list_departments",
            description="列出所有部门",
            inputSchema={
                "type": "object",
                "properties": {}
            }
        )
    ]

def load_employees() -> list[dict]:
    """加载员工数据"""
    if not EMPLOYEES_CSV.exists():
        return []

    employees = []
    with open(EMPLOYEES_CSV, encoding="utf-8") as f:
        reader = csv.DictReader(f)
        for row in reader:
            employees.append(row)

    return employees

async def handle_employee_tool(name: str, arguments: Any) -> list[TextContent]:
    employees = load_employees()

    if name == "search_employee":
        name_query = arguments.get("name", "").lower()
        dept_query = arguments.get("department", "").lower()

        results = [
            emp for emp in employees
            if (not name_query or name_query in emp.get("name", "").lower())
            and (not dept_query or dept_query in emp.get("department", "").lower())
        ]

        if not results:
            return [TextContent(type="text", text="未找到匹配的员工")]

        output = f"找到 {len(results)} 名员工：\n\n"
        for emp in results[:10]:
            output += f"👤 {emp.get('name', 'N/A')} | {emp.get('title', 'N/A')}\n"
            output += f"   部门：{emp.get('department', 'N/A')} | 邮箱：{emp.get('email', 'N/A')}\n\n"

        return [TextContent(type="text", text=output)]

    elif name == "get_department":
        dept = arguments["department"].lower()

        members = [
            emp for emp in employees
            if dept in emp.get("department", "").lower()
        ]

        if not members:
            return [TextContent(type="text", text=f"未找到部门：{arguments['department']}")]

        output = f"{arguments['department']} 部门成员（共 {len(members)} 人）：\n\n"
        for emp in members:
            output += f"👤 {emp.get('name', 'N/A')} | {emp.get('title', 'N/A')}\n"

        return [TextContent(type="text", text=output)]

    elif name == "list_departments":
        departments = set(emp.get("department", "") for emp in employees if emp.get("department"))

        if not departments:
            return [TextContent(type="text", text="暂无部门数据")]

        output = f"共有 {len(departments)} 个部门：\n\n"
        for dept in sorted(departments):
            count = sum(1 for emp in employees if emp.get("department") == dept)
            output += f"🏢 {dept}（{count} 人）\n"

        return [TextContent(type="text", text=output)]

    return [TextContent(type="text", text=f"Unknown tool: {name}")]
```

#### tools/project_tools.py

```python
import json
from pathlib import Path
from typing import Any
from mcp.types import Tool, TextContent

PROJECTS_JSON = Path(__file__).parent.parent / "data" / "projects.json"

def list_project_tools() -> list[Tool]:
    return [
        Tool(
            name="get_project",
            description="获取项目详情",
            inputSchema={
                "type": "object",
                "properties": {
                    "project_id": {
                        "type": "string",
                        "description": "项目 ID"
                    }
                },
                "required": ["project_id"]
            }
        ),
        Tool(
            name="list_projects",
            description="列出所有项目（可按状态过滤）",
            inputSchema={
                "type": "object",
                "properties": {
                    "status": {
                        "type": "string",
                        "description": "项目状态（active/completed/on-hold）",
                        "default": ""
                    },
                    "team": {
                        "type": "string",
                        "description": "团队名称",
                        "default": ""
                    }
                }
            }
        ),
        Tool(
            name="get_project_timeline",
            description="获取项目时间线",
            inputSchema={
                "type": "object",
                "properties": {
                    "project_id": {
                        "type": "string",
                        "description": "项目 ID"
                    }
                },
                "required": ["project_id"]
            }
        )
    ]

def load_projects() -> dict:
    """加载项目数据"""
    if not PROJECTS_JSON.exists():
        return {}

    with open(PROJECTS_JSON, encoding="utf-8") as f:
        return json.load(f)

async def handle_project_tool(name: str, arguments: Any) -> list[TextContent]:
    projects_data = load_projects()
    projects = projects_data.get("projects", [])

    if name == "get_project":
        project_id = arguments["project_id"]

        project = next((p for p in projects if p.get("id") == project_id), None)

        if not project:
            return [TextContent(type="text", text=f"未找到项目：{project_id}")]

        output = f"📋 {project.get('name', 'N/A')}\n\n"
        output += f"ID: {project.get('id')}\n"
        output += f"状态: {project.get('status', 'N/A')}\n"
        output += f"团队: {project.get('team', 'N/A')}\n"
        output += f"描述: {project.get('description', 'N/A')}\n"
        output += f"开始日期: {project.get('start_date', 'N/A')}\n"
        output += f"结束日期: {project.get('end_date', 'N/A')}\n"

        members = project.get("members", [])
        if members:
            output += f"成员: {', '.join(members)}\n"

        return [TextContent(type="text", text=output)]

    elif name == "list_projects":
        status_filter = arguments.get("status", "").lower()
        team_filter = arguments.get("team", "").lower()

        filtered = [
            p for p in projects
            if (not status_filter or status_filter in p.get("status", "").lower())
            and (not team_filter or team_filter in p.get("team", "").lower())
        ]

        if not filtered:
            return [TextContent(type="text", text="未找到匹配的项目")]

        status_icons = {
            "active": "🟢",
            "completed": "✅",
            "on-hold": "⏸️",
            "cancelled": "❌"
        }

        output = f"找到 {len(filtered)} 个项目：\n\n"
        for p in filtered:
            icon = status_icons.get(p.get("status", ""), "⚪")
            output += f"{icon} {p.get('name', 'N/A')}\n"
            output += f"   团队：{p.get('team', 'N/A')} | "
            output += f"状态：{p.get('status', 'N/A')}\n\n"

        return [TextContent(type="text", text=output)]

    elif name == "get_project_timeline":
        project_id = arguments["project_id"]

        project = next((p for p in projects if p.get("id") == project_id), None)

        if not project:
            return [TextContent(type="text", text=f"未找到项目：{project_id}")]

        milestones = project.get("milestones", [])

        if not milestones:
            return [TextContent(type="text", text=f"项目 {project_id} 暂无里程碑数据")]

        output = f"📅 {project.get('name', 'N/A')} 时间线\n\n"
        for m in milestones:
            status_icon = "✅" if m.get("completed") else "🔄"
            output += f"{status_icon} {m.get('date', 'N/A')}: {m.get('title', 'N/A')}\n"
            if m.get("description"):
                output += f"   {m.get('description')}\n"

        return [TextContent(type="text", text=output)]

    return [TextContent(type="text", text=f"Unknown tool: {name}")]
```
