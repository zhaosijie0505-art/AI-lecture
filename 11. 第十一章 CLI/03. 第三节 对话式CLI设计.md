# 第三节 对话式 CLI 设计

## 什么是对话式 CLI

传统 CLI 是"命令-响应"模式，用户输入完整命令，程序返回结果。

**对话式 CLI** 扩展为"多轮对话"模式，程序可以主动提问、记忆上下文、支持分支交互。

```bash
# 传统 CLI
$ deploy --env production --name my-app --image myregistry/app:v2
部署完成！

# 对话式 CLI
$ deploy
? 项目名称: my-app
? 选择环境 [dev/staging/prod]: production
? 镜像地址 (按 Enter 使用默认值 myregistry/app:latest): myregistry/app:v2
? 是否需要数据库？[y/N]: y

正在部署 my-app 到 production...
✓ 镜像拉取成功
✓ 数据库创建成功
✓ 应用启动成功
部署完成！
```

### 何时使用对话式 CLI

```
适合对话式：
- 参数较多，逐个确认更清晰
- 需要根据前面选择动态决定后续选项
- 首次使用，不熟悉命令参数
- 不频繁执行的操作（部署、初始化等）

不适合对话式：
- 自动化脚本（需要无人值守）
- 频繁重复的操作（管道组合）
- 参数简单明确的操作
```

## 实现方式

### 1. 简单实现：无限循环

```python
import click

def chat_loop():
    """简单的问答循环"""
    while True:
        query = input("你: ").strip()

        if query in ["退出", "exit", "quit"]:
            print("再见！")
            break

        if not query:
            continue

        response = process_query(query)
        print(f"AI: {response}")

def process_query(query):
    """处理用户输入"""
    # 这里是 AI 逻辑
    if "你好" in query:
        return "你好！有什么可以帮你？"
    elif "部署" in query:
        return "好的，请告诉我环境（dev/staging/prod）和应用名"
    else:
        return "我不明白，请再说一遍"

if __name__ == "__main__":
    chat_loop()
```

### 2. Click 的 prompt 模式

```python
import click

@click.command()
def init_project():
    """初始化项目（对话式）"""

    click.echo("=== 项目初始化向导 ===\n")

    # 必填参数
    name = click.prompt("项目名称", type=str)

    # 选择题
    framework = click.prompt(
        "框架",
        type=click.Choice(["FastAPI", "Django", "Flask", "None"]),
        default="FastAPI"
    )

    # 布尔确认
    use_db = click.confirm("是否启用数据库？", default=True)

    db_type = None
    if use_db:
        db_type = click.prompt(
            "数据库类型",
            type=click.Choice(["PostgreSQL", "MySQL", "SQLite"]),
            default="PostgreSQL"
        )

    # 可选参数
    port = click.prompt(
        "服务端口",
        type=int,
        default=8000,
        show_default=True
    )

    # 确认
    click.echo("\n=== 配置确认 ===")
    click.echo(f"项目名称: {name}")
    click.echo(f"框架: {framework}")
    click.echo(f"数据库: {db_type or '无'}")
```

### 3. Typer 的对话模式

```python
import typer
from typing import Optional

app = typer.Typer()

@app.command()
def create_project():
    """创建新项目（对话式）"""
    typer.echo("=== 创建新项目 ===\n")

    name = typer.prompt("项目名称")
    framework = typer.prompt(
        "框架",
        default="FastAPI",
        type=str
    )

    use_db = typer.confirm("启用数据库？", default=False)
    db_type = None
    if use_db:
        db_type = typer.prompt(
            "数据库类型",
            default="PostgreSQL",
            type=click.Choice(["PostgreSQL", "MySQL", "SQLite"])
        )

    typer.echo(f"\n✓ 创建项目: {name}（{framework}）")
    if db_type:
        typer.echo(f"✓ 数据库: {db_type}")

@app.command()
def deploy():
    """部署应用（对话式）"""
    typer.echo("=== 部署应用 ===\n")

    name = typer.prompt("应用名称")
    env = typer.prompt(
        "环境",
        type=typer.Choice(["dev", "staging", "prod"])
    )

    image = typer.prompt(
        "镜像地址",
        default=f"myregistry/{name}:latest"
    )

    if typer.confirm(f"确认部署 {name} 到 {env}？"):
        typer.echo(f"\n正在部署...")
        typer.echo(f"✓ 部署完成！")
```

## 状态管理

### 简单的状态字典

```python
import json
import os

STATE_FILE = ".cli_state.json"

def load_state():
    """加载状态"""
    if os.path.exists(STATE_FILE):
        with open(STATE_FILE) as f:
            return json.load(f)
    return {}

def save_state(state):
    """保存状态"""
    with open(STATE_FILE, "w") as f:
        json.dump(state, f, indent=2)

def chat_loop():
    state = load_state()
    typer.echo("欢迎回来！")

    # 恢复上下文
    if "current_project" in state:
        typer.echo(f"当前项目: {state['current_project']}")

    while True:
        query = typer.prompt("你")

        if query in ["退出", "exit"]:
            save_state(state)
            break

        # 处理命令，更新状态
        if query.startswith("切换项目"):
            project = query.replace("切换项目", "").strip()
            state["current_project"] = project
            typer.echo(f"已切换到项目: {project}")

        elif query == "当前项目":
            project = state.get("current_project", "无")
            typer.echo(f"当前项目: {project}")
```

### 分步骤流程

```python
import typer
from enum import Enum

class DeployStep(Enum):
    PROJECT = 1
    ENV = 2
    IMAGE = 3
    CONFIRM = 4
    DONE = 5

@app.command()
def deploy():
    """分步部署流程"""

    state = {
        "project": None,
        "env": None,
        "image": None
    }

    step = DeployStep.PROJECT

    while step != DeployStep.DONE:
        if step == DeployStep.PROJECT:
            project = typer.prompt("项目名称")
            state["project"] = project
            step = DeployStep.ENV

        elif step == DeployStep.ENV:
            env = typer.prompt(
                "环境 (dev/staging/prod)",
                type=typer.Choice(["dev", "staging", "prod"])
            )
            state["env"] = env

            # 根据环境动态调整默认值
            if env == "prod":
                typer.echo("⚠️  生产环境，请谨慎操作！")

            step = DeployStep.IMAGE

        elif step == DeployStep.IMAGE:
            default_image = f"myregistry/{state['project']}:latest"
            image = typer.prompt(
                "镜像地址",
                default=default_image
            )
            state["image"] = image
            step = DeployStep.CONFIRM

        elif step == DeployStep.CONFIRM:
            typer.echo("\n=== 部署确认 ===")
            for k, v in state.items():
                typer.echo(f"  {k}: {v}")

            if typer.confirm("确认部署？"):
                execute_deploy(state)
                typer.echo("✓ 部署完成！")
                step = DeployStep.DONE
            else:
                typer.echo("已取消")
                step = DeployStep.DONE
```

```python
from click.utils import echo
import itertools

def print_table(headers, rows):
    """打印表格"""
    col_widths = [len(h) for h in headers]

    for row in rows:
        for i, cell in enumerate(row):
            col_widths[i] = max(col_widths[i], len(str(cell)))

    # 表头
    header_line = " | ".join(h.ljust(w) for h, w in zip(headers, col_widths))
    separator = "-+-".join("-" * w for w in col_widths)

    click.echo(header_line)
    click.echo(separator)

    # 数据行
    for row in rows:
        line = " | ".join(str(cell).ljust(w) for cell, w in zip(row, col_widths))
        click.echo(line)

# 使用
print_table(["名称", "状态", "版本"],
            [["my-app", "running", "v1.2.0"],
             ["api", "stopped", "v2.0.0"]])
```

### 状态驱动的输出

```python
from enum import Enum

class Status(Enum):
    SUCCESS = ("✓", "green")
    ERROR = ("✗", "red")
    WARNING = ("⚠", "yellow")
    INFO = ("ℹ", "blue")
    LOADING = ("⟳", "cyan")

    def __init__(self, icon, color):
        self.icon = icon
        self.color = color

def log(status: Status, message: str):
    """统一日志输出"""
    icon = click.style(status.icon, fg=status.color)
    click.echo(f"{icon} {message}")

def section(title: str):
    """章节标题"""
    click.echo(f"\n{click.style('─' * 40, fg='dim')}")
    click.echo(click.style(f" {title}", bold=True))

# 使用
section("部署开始")
log(Status.INFO, "检查镜像...")
log(Status.SUCCESS, "镜像存在")
log(Status.LOADING, "正在拉取...")
log(Status.SUCCESS, "镜像拉取完成")
log(Status.WARNING, "生产环境确认")
log(Status.ERROR, "部署失败：端口被占用")
```

## 本节小结

**对话式 CLI** 在传统"命令-响应"基础上增加了多轮交互、状态管理、分步流程。

**实现方式**：
- 简单循环：`while True` + `input()`
- Click prompt 模式：`click.prompt()`、`click.confirm()`
- Typer prompt 模式：`typer.prompt()`、`typer.confirm()`

**状态管理**：
- 简单状态：`dict` 存储
- 分步流程：状态机（`Enum` + `step` 变量）
- 持久化：`JSON` 文件

**最佳实践**：
- 必填参数用 prompt，可选参数用默认值
- 敏感操作要确认（`confirm`）
- 用颜色区分状态（成功/警告/错误）
- 进度条改善长任务体验
