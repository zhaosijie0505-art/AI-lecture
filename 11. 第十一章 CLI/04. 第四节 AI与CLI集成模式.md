# 第四节 AI 与 CLI 集成模式

## 核心问题

AI 的输出是自然语言，CLI 的输入是结构化命令。**集成模式**解决的是：如何把 AI 的"想法"变成可执行的 CLI 命令，并解析执行结果。

```
集成流程：

用户：帮我把这个目录下的所有 txt 文件压缩成 archive.zip

    ┌──────────────────────────────────────┐
    │           AI 分析                     │
    │  "用户要压缩文件"                      │
    │  "目标目录：当前"                      │
    │  "模式：所有 .txt 文件"                │
    │  "输出：archive.zip"                  │
    └──────────────┬───────────────────────┘
                   │ 生成命令
                   ▼
    ┌──────────────────────────────────────┐
    │           命令生成                     │
    │  zip archive.zip *.txt               │
    └──────────────┬───────────────────────┘
                   │ 执行
                   ▼
    ┌──────────────────────────────────────┐
    │           CLI 执行                    │
    │  $ zip archive.zip *.txt             │
    │  adding: file1.txt                   │
    │  adding: file2.txt                   │
    └──────────────┬───────────────────────┘
                   │ 解析输出
                   ▼
    ┌──────────────────────────────────────┐
    │           结果反馈                     │
    │  "已压缩 2 个文件为 archive.zip"       │
    └──────────────────────────────────────┘
```

## 模式一：管道模式（Pipe）

### 原理

AI 生成命令字符串，通过 `subprocess` 执行，捕获输出。

```python
import subprocess
import shlex

def execute_command(cmd: str) -> tuple[int, str, str]:
    """
    执行命令，返回 (exit_code, stdout, stderr)
    """
    try:
        result = subprocess.run(
            cmd,
            shell=True,              # 允许 shell 语法
            capture_output=True,     # 捕获输出
            text=True,               # 返回字符串
            timeout=30
        )
        return result.returncode, result.stdout, result.stderr
    except subprocess.TimeoutExpired:
        return -1, "", "命令执行超时"
    except Exception as e:
        return -1, "", str(e)

def ai_execute(user_request: str, llm) -> str:
    """AI 驱动的命令执行"""

    # Step 1: 让 LLM 生成命令
    prompt = f"""用户请求: {user_request}

请生成对应的 shell 命令。只输出命令，不要其他解释。
如果需要多步操作，用 && 连接。
"""

    command = llm.complete(prompt).strip()

    # Step 2: 执行命令
    exit_code, stdout, stderr = execute_command(command)

    # Step 3: 格式化结果
    if exit_code == 0:
        return f"✓ 执行成功\n\n输出:\n{stdout or '(无输出)'}"
    else:
        return f"✗ 执行失败 (exit code: {exit_code})\n\n错误:\n{stderr}"

# 使用
result = ai_execute("查看当前目录的 txt 文件", llm)
print(result)
```

### 带安全检查

```python
import subprocess
import shlex

# 危险命令黑名单
DANGEROUS_COMMANDS = ["rm -rf /", "rm -rf /*", "mkfs", ":(){ :|:& };:"]

def is_safe_command(cmd: str) -> bool:
    """检查命令安全性"""
    cmd_lower = cmd.lower()

    # 检查危险命令
    for dangerous in DANGEROUS_COMMANDS:
        if dangerous in cmd_lower:
            return False

    # 检查是否有 ; 或 && 链接多条命令
    # （可根据需要调整策略）
    if "&&" in cmd or ";" in cmd:
        # 可以添加额外检查
        pass

    return True

def safe_execute(cmd: str) -> str:
    """安全的命令执行"""

    if not is_safe_command(cmd):
        return "✗ 命令被安全策略拦截（可能包含危险操作）"

    exit_code, stdout, stderr = execute_command(cmd)

    if exit_code == 0:
        return f"✓ 成功\n{stdout or '(无输出)'}"
    else:
        return f"✗ 失败 (code {exit_code})\n{stderr}"

def ai_shell(user_request: str, llm) -> str:
    """AI Shell 助手"""

    # 生成命令
    command = llm.complete(
        f"为这个请求生成 shell 命令：{user_request}"
    ).strip()

    # 显示将要执行的命令，等待确认
    print(f"将要执行: {command}")

    # 自动执行（或者等待用户确认）
    return safe_execute(command)
```

## 模式二：Structured Output（推荐）

### 原理

让 AI 输出 JSON 结构化数据，明确命令、解释、预期结果。

```python
import subprocess
import json
from typing import Optional

# 定义输出格式
COMMAND_SCHEMA = {
    "type": "object",
    "properties": {
        "command": {
            "type": "string",
            "description": "要执行的完整命令"
        },
        "explanation": {
            "type": "string",
            "description": "命令的解释"
        },
        "warning": {
            "type": "string",
            "description": "警告信息（如果有）"
        }
    },
    "required": ["command"]
}

def execute_shell(command: str, timeout: int = 30) -> dict:
    """执行命令并返回结构化结果"""
    try:
        result = subprocess.run(
            command,
            shell=True,
            capture_output=True,
            text=True,
            timeout=timeout
        )
        return {
            "success": result.returncode == 0,
            "exit_code": result.returncode,
            "stdout": result.stdout,
            "stderr": result.stderr
        }
    except subprocess.TimeoutExpired:
        return {
            "success": False,
            "exit_code": -1,
            "stdout": "",
            "stderr": "执行超时"
        }

def ai_command(user_request: str, llm) -> str:
    """AI 生成结构化命令并执行"""

    # 生成 JSON 格式的命令
    prompt = f"""用户请求：{user_request}

生成对应的 shell 命令。响应格式：
{{
  "command": "实际命令",
  "explanation": "命令解释",
  "warning": "警告信息（无则省略）"
}}
"""

    response = llm.complete(prompt)

    # 解析 JSON
    try:
        data = json.loads(response)
    except:
        return f"无法解析命令响应：{response}"

    command = data["command"]
    explanation = data.get("explanation", "")

    if explanation:
        print(f"解释: {explanation}")

    if data.get("warning"):
        print(f"⚠️  警告: {data['warning']}")

    # 执行
    result = execute_shell(command)

    # 格式化结果
    if result["success"]:
        return f"✓ 成功\n\n{result['stdout'] or '命令执行成功（无输出）'}"
    else:
        return f"✗ 失败 (exit code: {result['exit_code']})\n\n错误:\n{result['stderr']}"

# 使用示例
# result = ai_command("列出当前目录的所有 Python 文件", llm)
```

## 模式三：解析与重组

### 解析 CLI 输出

CLI 输出的文本需要被 AI 理解和处理：

```python
import re
import json

def parse_table_output(text: str) -> list[dict]:
    """解析表格输出"""
    lines = text.strip().split("\n")

    if len(lines) < 2:
        return []

    # 假设第一行是表头
    headers = lines[0].split()

    rows = []
    for line in lines[1:]:
        cells = line.split()
        if len(cells) == len(headers):
            rows.append(dict(zip(headers, cells)))

    return rows

def parse_json_output(text: str) -> Optional[dict]:
    """解析 JSON 输出"""
    try:
        # 尝试提取 JSON
        text = text.strip()
        if text.startswith("{"):
            return json.loads(text)
        # 可能包裹在其他文本中
        match = re.search(r'\{[^{}]*\}', text)
        if match:
            return json.loads(match.group())
    except:
        pass
    return None

def parse_kv_output(text: str) -> dict:
    """解析 key=value 输出"""
    result = {}
    for line in text.strip().split("\n"):
        if "=" in line:
            key, value = line.split("=", 1)
            result[key.strip()] = value.strip()
    return result

# 使用示例
# docker ps 输出解析
docker_ps = """CONTAINER ID   IMAGE          STATUS       PORTS
abc123        nginx:latest   Up 2 hours   0.0.0.0:80->80/tcp
def456        redis:alpine   Up 5 days    6379/tcp"""

containers = parse_table_output(docker_ps)
for c in containers:
    print(f"容器: {c.get('IMAGE', 'N/A')} - {c.get('STATUS', 'N/A')}")
```

## 管道组合实践

### 管道设计原则

```
1. 每个命令只做一件事（Unix 哲学）
2. 输出可以被下一个命令消费
3. 错误处理要明确

示例：统计项目代码行数

git ls-files | grep "\.py$" | xargs wc -l | sort -rn

分解：
git ls-files           → 列出所有受版本控制的文件
grep "\.py$"           → 过滤 Python 文件
xargs wc -l            → 统计每个文件的行数
sort -rn               → 按行数降序排序
```

### AI 生成管道

```python
def generate_pipeline(request: str, llm) -> list[str]:
    """AI 生成管道命令"""
    prompt = f"""用户需求：{request}

请分析需求，生成一系列管道命令（每行一个命令）。
只输出命令，不要其他解释。

需求可能涉及：
- 文件查找/过滤
- 数据处理
- 统计/排序
- 输出格式化

示例输出：
find . -name "*.py"
xargs wc -l
sort -rn
"""

    response = llm.complete(prompt)

    # 解析命令列表
    commands = [line.strip() for line in response.split("\n") if line.strip()]

    return commands

def execute_pipeline(commands: list[str]) -> str:
    """执行管道命令序列"""
    if not commands:
        return "无命令可执行"

    # 简化实现：逐个执行，返回最后输出
    last_output = ""

    for cmd in commands:
        exit_code, stdout, stderr = execute_command(cmd)
        last_output = stdout if exit_code == 0 else stderr

        if exit_code != 0:
            return f"命令失败: {cmd}\n错误: {stderr}"

    return last_output
```

## 本节小结

**三大集成模式**：

1. **管道模式（Pipe）**
   - AI 生成命令 → `subprocess.run()` 执行 → 返回结果
   - 简单直接，但缺乏结构

2. **Structured Output（推荐）**
   - AI 输出 JSON（command + explanation + warning）
   - 明确解析，结构清晰，可扩展

3. **解析与重组**
   - 解析 CLI 输出（表格/JSON/KV）
   - 将结果注入 AI 上下文做进一步处理

**安全策略**：
- 黑名单过滤危险命令
- 执行前用户确认
- 超时控制

**实践要点**：
- AI 输出要明确（Structured Output）
- 错误信息要保留（stderr）
- 管道组合遵循 Unix 哲学
