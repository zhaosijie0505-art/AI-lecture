# 第五节 Shell Agent 实战

## 什么是 Shell Agent

**Shell Agent** 是一个 AI Agent，它能理解自然语言命令，通过 CLI 执行实际操作，形成"思考→执行→反馈"的闭环。

```
传统 CLI：
用户 → 输入完整命令 → 执行 → 看结果

Shell Agent：
用户（自然语言）→ AI 理解意图 → 生成命令 → 执行 → 解析结果 → 反馈
```

### Shell 执行器（shell.py）

```python
import subprocess
import json
import os
from typing import Optional
from dataclasses import dataclass

@dataclass
class CommandResult:
    """命令执行结果"""
    command: str
    exit_code: int
    stdout: str
    stderr: str
    duration: float
    success: bool

class ShellExecutor:
    """Shell 命令执行器"""

    def __init__(self, config):
        self.config = config
        self.history = []

    def execute(self, command: str, timeout: int = None) -> CommandResult:
        """执行命令，返回结果"""
        import time

        if timeout is None:
            timeout = self.config.TIMEOUT

        # 安全检查
        if not self._is_safe(command):
            return CommandResult(
                command=command,
                exit_code=-1,
                stdout="",
                stderr="命令被安全策略拦截",
                duration=0,
                success=False
            )

        start = time.time()

        try:
            result = subprocess.run(
                command,
                shell=True,
                capture_output=True,
                text=True,
                timeout=timeout,
                cwd=os.getcwd()  # 在当前目录执行
            )

            duration = time.time() - start

            cmd_result = CommandResult(
                command=command,
                exit_code=result.returncode,
                stdout=result.stdout,
                stderr=result.stderr,
                duration=duration,
                success=result.returncode == 0
            )

        except subprocess.TimeoutExpired:
            duration = time.time() - start
            cmd_result = CommandResult(
                command=command,
                exit_code=-1,
                stdout="",
                stderr=f"命令执行超时（{timeout}秒）",
                duration=duration,
                success=False
            )

        except Exception as e:
            duration = time.time() - start
            cmd_result = CommandResult(
                command=command,
                exit_code=-1,
                stdout="",
                stderr=str(e),
                duration=duration,
                success=False
            )

        # 记录历史
        self.history.append(cmd_result)
        self._save_history()

        return cmd_result

    def _is_safe(self, command: str) -> bool:
        """安全检查"""
        cmd_lower = command.lower()

        # 检查黑名单
        for blocked in self.config.BLOCKED_COMMANDS:
            if blocked in cmd_lower:
                return False

        return True

    def needs_confirmation(self, command: str) -> bool:
        """检查是否需要确认"""
        cmd_lower = command.lower()
        for pattern in self.config.REQUIRE_CONFIRMATION_PATTERNS:
            if pattern in cmd_lower:
                return True
        return False

    def _save_history(self):
        """保存历史记录"""
        try:
            with open(self.config.HISTORY_FILE, "w") as f:
                json.dump([
                    {
                        "command": r.command,
                        "exit_code": r.exit_code,
                        "stdout": r.stdout[:500],  # 截断
                        "duration": r.duration
                    }
                    for r in self.history[-100:]  # 只保留最近100条
                ], f, indent=2)
        except:
            pass
```

### 工具定义（tools.py）

```python
import os
from typing import TypeVar, Generic, Optional
from dataclasses import dataclass

@dataclass
class Tool:
    """工具定义"""
    name: str
    description: str
    input_schema: dict

# 可用的内置工具
BUILTIN_TOOLS = [
    Tool(
        name="shell",
        description="执行 shell 命令",
        input_schema={
            "type": "object",
            "properties": {
                "command": {"type": "string", "description": "要执行的命令"},
                "timeout": {"type": "number", "description": "超时秒数"}
            },
            "required": ["command"]
        }
    ),
    Tool(
        name="read_file",
        description="读取文件内容",
        input_schema={
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "文件路径"},
                "lines": {"type": "number", "description": "最大行数"}
            },
            "required": ["path"]
        }
    ),
    Tool(
        name="list_dir",
        description="列出目录内容",
        input_schema={
            "type": "object",
            "properties": {
                "path": {"type": "string", "description": "目录路径"}
            }
        }
    ),
    Tool(
        name="search_files",
        description="搜索文件内容",
        input_schema={
            "type": "object",
            "properties": {
                "pattern": {"type": "string", "description": "搜索内容"},
                "path": {"type": "string", "description": "搜索目录"},
                "file_type": {"type": "string", "description": "文件类型如 py/txt"}
            }
        }
    )
]

def execute_tool(name: str, args: dict, executor) -> str:
    """执行工具调用"""
    if name == "shell":
        result = executor.execute(args["command"], args.get("timeout"))
        if result.success:
            return f"✓ 执行成功 ({result.duration:.2f}s)\n\n{result.stdout or '(无输出)'}"
        else:
            return f"✗ 执行失败 (exit code: {result.exit_code})\n\n{result.stderr}"

    elif name == "read_file":
        path = args["path"]
        max_lines = args.get("lines", 100)

        if not os.path.exists(path):
            return f"文件不存在: {path}"

        with open(path) as f:
            lines = f.readlines()[:max_lines]
            content = "".join(lines)

            if len(lines) >= max_lines:
                content += f"\n... (共 {len(lines)}+ 行，已截断)"

        return content

    elif name == "list_dir":
        path = args.get("path", ".")
        if not os.path.exists(path):
            return f"目录不存在: {path}"

        items = []
        for item in os.listdir(path):
            full_path = os.path.join(path, item)
            icon = "📁" if os.path.isdir(full_path) else "📄"
            items.append(f"{icon} {item}")

        return "\n".join(items) if items else "目录为空"

    elif name == "search_files":
        pattern = args["pattern"]
        search_path = args.get("path", ".")
        file_type = args.get("file_type", "")

        results = []
        import glob

        pattern_glob = f"**/*.{file_type}" if file_type else "**/*"

        for file_path in glob.glob(f"{search_path}/{pattern_glob}", recursive=True):
            try:
                with open(file_path) as f:
                    content = f.read()
                    if pattern in content:
                        # 找到匹配行
                        for i, line in enumerate(f.readlines(), 1):
                            if pattern in line:
                                results.append(f"{file_path}:{i}: {line.strip()}")
            except:
                pass

        if not results:
            return f"未找到包含 '{pattern}' 的文件"

        return f"找到 {len(results)} 个匹配：\n\n" + "\n".join(results[:50])

    return f"未知工具: {name}"
```

```python
    prompt = request
    if context:
        prompt = f"{context}\n\n---\n\n用户请求: {request}"
    return prompt
```

### Agent 核心（agent.py）

```python
import json
from typing import Optional
from dataclasses import dataclass

@dataclass
class Message:
    """对话消息"""
    role: str  # system / user / assistant / tool
    content: str
    tool_calls: Optional[list] = None
    tool_results: Optional[list] = None

class ShellAgent:
    """Shell Agent 核心"""

    def __init__(self, executor, tools, config):
        self.executor = executor
        self.tools = tools
        self.config = config
        self.messages = []
        self.iteration = 0

    def chat(self, user_request: str) -> str:
        """处理用户请求"""
        self.iteration = 0

        # 添加用户消息
        self.messages.append(Message(role="user", content=user_request))

        # 对话循环
        while self.iteration < self.config.MAX_ITERATIONS:
            self.iteration += 1

            # 调用 LLM
            response = self._call_llm()

            # 没有 tool_calls，结束
            if not response.tool_calls:
                if response.content:
                    self.messages.append(Message(role="assistant", content=response.content))
                    return response.content
                break

            # 处理 tool_calls
            for tool_call in response.tool_calls:
                tool_name = tool_call.name
                tool_args = tool_call.arguments

                # 执行工具
                result = self._call_tool(tool_name, tool_args)

                self.messages.append(Message(
                    role="assistant",
                    content="",
                    tool_calls=[tool_call]
                ))
                self.messages.append(Message(
                    role="tool",
                    content=result,
                    tool_results=[{"name": tool_name, "result": result}]
                ))

        return "已达到最大迭代次数"
```

### 主入口（main.py）

```python
#!/usr/bin/env python3
"""
Shell Agent - 自然语言驱动的命令行助手
"""

import typer

app = typer.Typer()

@app.command()
def main(
    request: str = typer.Argument(None, help="要执行的请求"),
    dry_run: bool = typer.Option(False, "--dry-run", help="仅模拟，不执行"),
    verbose: bool = typer.Option(False, "--verbose", "-v", help="详细输出"),
    interactive: bool = typer.Option(False, "--interactive", "-i", help="交互模式"),
):
    """Shell Agent - 自然语言驱动的命令行助手"""

    from config import Config
    from shell import ShellExecutor
    from tools import BUILTIN_TOOLS, execute_tool
    from agent import ShellAgent

    cfg = Config()
    if dry_run:
        cfg.DRY_RUN = True
        cfg.ALLOW_EXECUTION = False

    executor = ShellExecutor(cfg)
    agent = ShellAgent(executor, BUILTIN_TOOLS, cfg)

    if interactive:
        # 交互模式
        typer.echo("Shell Agent 已启动（输入 'exit' 退出）\n")

        while True:
            try:
                user_input = typer.prompt("你")
                if user_input.lower() in ["exit", "quit", "退出"]:
                    break

                result = agent.chat(user_input)
                typer.echo(f"\n{result}\n")

            except KeyboardInterrupt:
                typer.echo("\n\n退出")
                break

    elif request:
        # 单次请求模式
        result = agent.chat(request)
        typer.echo(result)

    else:
        typer.echo("请提供请求，或使用 --interactive 进入交互模式")

if __name__ == "__main__":
    app()
```

## 本节小结

**Shell Agent** 把自然语言请求转化为可执行的 CLI 命令，形成"理解→执行→反馈"的闭环。

**核心组件**：
- **ShellExecutor**：命令执行器（安全检查、超时、history）
- **Tools**：可用工具（shell/read_file/list_dir/search_files）
- **Agent**：对话循环（LLM 调用 + 工具调度）
- **Prompts**：Prompt 模板（系统指令 + 用户请求）

**安全策略**：
- 黑名单过滤危险命令
- 破坏性操作需要确认
- 超时控制
- 执行 history 记录

**扩展方向**：
- 接入真实 LLM（OpenAI/Anthropic API）
- 增加更多工具（git/docker/kubectl）
- 支持多轮对话上下文
- 会话持久化
